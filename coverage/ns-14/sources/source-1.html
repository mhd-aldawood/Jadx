417 	private ArgType readArgType(DataInputStream in) throws IOException {
418 		int ordinal = in.readByte();
419 		if (ordinal == -1) {
420 			return null;
421 		}
422 		if (ordinal >= TypeEnum.values().length) {
423 			throw new JadxRuntimeException("Incorrect ordinal for type enum: " + ordinal);
424 		}
425 		switch (TypeEnum.values()[ordinal]) {
426 			case WILDCARD:
427 				ArgType.WildcardBound bound = ArgType.WildcardBound.getByNum(in.readByte());
428 				if (bound == ArgType.WildcardBound.UNBOUND) {
429 					return ArgType.WILDCARD;
430 				}
431 				ArgType objType = readArgType(in);
432 				return ArgType.wildcard(objType, bound);
433 
434 			case OUTER_GENERIC:
435 				ArgType outerType = readArgType(in);
436 				ArgType innerType = readArgType(in);
437 				return ArgType.outerGeneric(outerType, innerType);
438 
439 			case GENERIC:
440 				ArgType clsType = classes[in.readInt()].getClsType();
441 				return ArgType.generic(clsType, readArgTypesList(in));
442 
443 			case GENERIC_TYPE_VARIABLE:
444 				String typeVar = readString(in);
445 				List<ArgType> extendTypes = readArgTypesList(in);
446 				return ArgType.genericType(typeVar, extendTypes);
447 
448 			case OBJECT:
449 				return classes[in.readInt()].getClsType();
450 
451 			case ARRAY:
452 				return ArgType.array(readArgType(in));
453 
454 			case PRIMITIVE:
455 				char shortName = (char) in.readByte();
456 				return ArgType.parse(shortName);
457 
458 			default:
459 				throw new JadxRuntimeException("Unsupported Arg Type: " + ordinal);
460 		}
461 	}
