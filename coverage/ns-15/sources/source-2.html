441 		EnumClassAttr enumFields = cls.get(AType.ENUM_CLASS);
442 		if (enumFields == null) {
443 			return;
444 		}
445 		InsnGen igen = null;
446 		for (Iterator<EnumField> it = enumFields.getFields().iterator(); it.hasNext();) {
447 			EnumField f = it.next();

453 			int skipCount = getEnumCtrSkipArgsCount(callMth);
454 			if (constrInsn.getArgsCount() > skipCount) {

457 				}

463 			}
464 			if (it.hasNext()) {
465 				code.add(',');
466 			}
467 		}

469 			if (enumFields.getFields().isEmpty()) {
470 				code.startLine();
471 			}
472 			code.add(';');

565 		List<ArgType> generics = type.getGenericTypes();
566 		if (generics != null) {
567 			code.add('<');
568 			int len = generics.size();
569 			for (int i = 0; i < len; i++) {
570 				if (i != 0) {
571 					code.add(", ");
572 				}
573 				ArgType gt = generics.get(i);
574 				ArgType wt = gt.getWildcardType();
575 				if (wt != null) {
576 					ArgType.WildcardBound bound = gt.getWildcardBound();
577 					code.add(bound.getStr());
578 					if (bound != ArgType.WildcardBound.UNBOUND) {
579 						useType(code, wt);
580 					}
581 				} else {

608 	private String useClassInternal(ClassInfo useCls, ClassInfo extClsInfo) {
609 		String fullName = extClsInfo.getAliasFullName();
610 		if (fallback || !useImports) {
611 			return fullName;
612 		}
613 		String shortName = extClsInfo.getAliasShortName();
614 		if (extClsInfo.getPackage().equals("java.lang") && extClsInfo.getParentClass() == null) {
615 			return shortName;
616 		}
617 		if (isClassInnerFor(useCls, extClsInfo)) {
618 			return shortName;
619 		}
620 		if (extClsInfo.isInner()) {
621 			return expandInnerClassName(useCls, extClsInfo);
622 		}
623 		if (isBothClassesInOneTopClass(useCls, extClsInfo)) {
624 			return shortName;
625 		}
626 		// don't add import if this class from same package
627 		if (extClsInfo.getPackage().equals(useCls.getPackage()) && !extClsInfo.isInner()) {
628 			return shortName;
629 		}
630 		if (searchCollision(cls.root(), useCls, extClsInfo)) {
631 			return fullName;
632 		}
633 		// ignore classes from default package
634 		if (extClsInfo.isDefaultPackage()) {
635 			return shortName;
636 		}
637 		if (extClsInfo.getAliasPkg().equals(useCls.getAliasPkg())) {
638 			fullName = extClsInfo.getAliasNameWithoutPackage();
639 		}
640 		for (ClassInfo importCls : getImports()) {
641 			if (!importCls.equals(extClsInfo)
642 					&& importCls.getAliasShortName().equals(shortName)) {
643 				if (extClsInfo.isInner()) {
644 					String parent = useClassInternal(useCls, extClsInfo.getParentClass());
645 					return parent + '.' + shortName;
646 				} else {
647 					return fullName;
648 				}
649 			}
650 		}
651 		addImport(extClsInfo);
652 		return shortName;
653 	}

655 	private String expandInnerClassName(ClassInfo useCls, ClassInfo extClsInfo) {
656 		List<ClassInfo> clsList = new ArrayList<>();
657 		clsList.add(extClsInfo);
658 		ClassInfo parentCls = extClsInfo.getParentClass();
659 		boolean addImport = true;
660 		while (parentCls != null) {
661 			if (parentCls == useCls || isClassInnerFor(useCls, parentCls)) {
662 				addImport = false;
663 				break;
664 			}
665 			clsList.add(parentCls);
666 			parentCls = parentCls.getParentClass();
667 		}
668 		Collections.reverse(clsList);
669 		if (addImport) {
670 			addImport(clsList.get(0));
671 		}
672 		return Utils.listToString(clsList, ".", ClassInfo::getAliasShortName);
673 	}

709 	private static boolean searchCollision(RootNode root, ClassInfo useCls, ClassInfo searchCls) {
710 		if (useCls == null) {
711 			return false;
712 		}
713 		String shortName = searchCls.getAliasShortName();
714 		if (useCls.getAliasShortName().equals(shortName)) {
715 			return true;
716 		}
717 		ClassNode classNode = root.resolveClass(useCls);
718 		if (classNode != null) {
719 			for (ClassNode inner : classNode.getInnerClasses()) {
720 				if (inner.getShortName().equals(shortName)
721 						&& !inner.getFullName().equals(searchCls.getAliasFullName())) {
722 					return true;
723 				}
724 			}
725 		}
726 		return searchCollision(root, useCls.getParentClass(), searchCls);
727 	}
