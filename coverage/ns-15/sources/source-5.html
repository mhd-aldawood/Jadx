248 	protected void makeInsn(InsnNode insn, ICodeWriter code, Flags flag) throws CodegenException {
249 		if (insn.getType() == InsnType.REGION_ARG) {
250 			return;
251 		}
252 		try {
253 			if (flag == Flags.BODY_ONLY || flag == Flags.BODY_ONLY_NOWRAP) {
254 				makeInsnBody(code, insn, flag == Flags.BODY_ONLY ? BODY_ONLY_FLAG : BODY_ONLY_NOWRAP_FLAGS);
255 			} else {
256 				if (flag != Flags.INLINE) {
257 					code.startLineWithNum(insn.getSourceLine());
258 					InsnCodeOffset.attach(code, insn);
259 					if (insn.contains(AFlag.COMMENT_OUT)) {
260 						code.add("// ");
261 					}
262 				}
263 				RegisterArg resArg = insn.getResult();
264 				if (resArg != null) {
265 					SSAVar var = resArg.getSVar();
266 					if (var == null || var.getUseCount() != 0 || insn.getType() != InsnType.CONSTRUCTOR) {
267 						assignVar(code, insn);
268 						code.add(" = ");
269 					}
270 				}
271 				makeInsnBody(code, insn, EMPTY_FLAGS);
272 				if (flag != Flags.INLINE) {
273 					code.add(';');
274 					CodeGenUtils.addCodeComments(code, mth, insn);
275 				}
276 			}
277 		} catch (Exception e) {
278 			throw new CodegenException(mth, "Error generate insn: " + insn, e);
279 		}
280 	}

674 		ClassNode cls = mth.root().resolveClass(insn.getClassType());
675 		if (cls != null && cls.isAnonymous() && !fallback) {
676 			cls.ensureProcessed();

680 		}

683 		}

700 			GenericInfoAttr genericInfoAttr = insn.get(AType.GENERIC_INFO);
701 			if (genericInfoAttr != null) {

705 					for (ArgType type : genericInfoAttr.getGenericTypes()) {
706 						if (!first) {
707 							code.add(',');
708 						} else {
709 							first = false;
710 						}
711 						mgen.getClassGen().useType(code, type);
712 					}
713 				}

715 			}

751 		}

760 	private void makeInvoke(InvokeNode insn, ICodeWriter code) throws CodegenException {
761 		InvokeType type = insn.getInvokeType();
762 		if (type == InvokeType.CUSTOM) {
763 			makeInvokeLambda(code, (InvokeCustomNode) insn);
764 			return;
765 		}
766 		MethodInfo callMth = insn.getCallMth();
767 		MethodNode callMthNode = mth.root().resolveMethod(callMth);
768 
769 		int k = 0;
770 		switch (type) {
771 			case DIRECT:
772 			case VIRTUAL:
773 			case INTERFACE:
774 				InsnArg arg = insn.getArg(0);
775 				if (needInvokeArg(arg)) {
776 					addArgDot(code, arg);
777 				}
778 				k++;
779 				break;
780 
781 			case SUPER:
782 				ClassInfo superCallCls = getClassForSuperCall(code, callMth);
783 				if (superCallCls != null) {
784 					useClass(code, superCallCls);
785 					code.add('.');
786 				}
787 				// use 'super' instead 'this' in 0 arg
788 				code.add("super").add('.');
789 				k++;
790 				break;
791 
792 			case STATIC:
793 				ClassInfo insnCls = mth.getParentClass().getClassInfo();
794 				ClassInfo declClass = callMth.getDeclClass();
795 				if (!insnCls.equals(declClass)) {
796 					useClass(code, declClass);
797 					code.add('.');
798 				}
799 				break;
800 		}
801 		if (callMthNode != null) {
802 			code.attachAnnotation(callMthNode);
803 			code.add(callMthNode.getAlias());
804 		} else {
805 			code.add(callMth.getAlias());
806 		}
807 		generateMethodArguments(code, insn, k, callMthNode);
808 	}

941 		ClassNode useCls = mth.getParentClass();
942 		ClassInfo insnCls = useCls.getClassInfo();

944 		if (insnCls.equals(declClass)) {

949 			return declClass;
