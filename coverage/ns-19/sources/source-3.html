419 	 * Generate a prefix for a class name that bases on certain class properties, certain
420 	 * extended superclasses or implemented interfaces.
421 	 */
422 	private String makeClsPrefix(ClassNode cls) {
423 		if (cls.isEnum()) {
424 			return "Enum";
425 		}
426 		String result = "";
427 		if (cls.getAccessFlags().isAbstract()) {
428 			result += "Abstract";
429 		}
430 
431 		// Process current class and all super classes
432 		ClassNode currentCls = cls;
433 		outerLoop: while (currentCls != null) {
434 			if (currentCls.getSuperClass() != null) {
435 				String superClsName = currentCls.getSuperClass().getObject();
436 				if (superClsName.startsWith("android.app.")) {
437 					// e.g. Activity or Fragment
438 					result += superClsName.substring(12);
439 					break;
440 				} else if (superClsName.startsWith("android.os.")) {
441 					// e.g. AsyncTask
442 					result += superClsName.substring(11);
443 					break;
444 				}
445 			}
446 			for (ArgType intf : cls.getInterfaces()) {
447 				String intfClsName = intf.getObject();
448 				if (intfClsName.equals("java.lang.Runnable")) {
449 					result += "Runnable";
450 					break outerLoop;
451 				} else if (intfClsName.startsWith("java.util.concurrent.")) {
452 					// e.g. Callable
453 					result += intfClsName.substring(21);
454 					break outerLoop;
455 				} else if (intfClsName.startsWith("android.view.")) {
456 					// e.g. View.OnClickListener
457 					result += intfClsName.substring(13);
458 					break outerLoop;
459 				} else if (intfClsName.startsWith("android.content.")) {
460 					// e.g. DialogInterface.OnClickListener
461 					result += intfClsName.substring(16);
462 					break outerLoop;
463 				}
464 			}
465 			if (currentCls.getSuperClass() == null) {
466 				break;
467 			}
468 			currentCls = cls.root().resolveClass(currentCls.getSuperClass());
469 		}
470 		return result;
471 	}

475 		SourceFileAttr sourceFileAttr = cls.get(JadxAttrType.SOURCE_FILE);
476 		if (sourceFileAttr == null) {
477 			return null;
478 		}
479 		if (cls.getClassInfo().isInner()) {
480 			return null;
481 		}
482 		String name = sourceFileAttr.getFileName();
483 		if (name.endsWith(".java")) {
484 			name = name.substring(0, name.length() - ".java".length());
485 		} else if (name.endsWith(".kt")) {
486 			name = name.substring(0, name.length() - ".kt".length());
487 		}
488 		if (!NameMapper.isValidAndPrintable(name)) {
489 			return null;
490 		}
491 		for (DeobfClsInfo deobfClsInfo : clsMap.values()) {
492 			if (deobfClsInfo.getAlias().equals(name)) {
493 				return null;
494 			}
495 		}
496 		ClassNode otherCls = cls.root().resolveClass(cls.getPackage() + '.' + name);
497 		if (otherCls != null) {
498 			return null;
499 		}
500 		cls.remove(JadxAttrType.SOURCE_FILE);
501 		return name;
502 	}

566 	private void processPackageFull(PackageNode pkg, String fullName) {
567 		if (pkgSet.contains(fullName)) {
568 			return;
569 		}
570 		pkgSet.add(fullName);
571 
572 		// doPkg for all parent packages except root that not hasAliases
573 		PackageNode parentPkg = pkg.getParentPackage();
574 		while (!parentPkg.getName().isEmpty()) {
575 			if (!parentPkg.hasAlias()) {
576 				processPackageFull(parentPkg, parentPkg.getFullName());
577 			}
578 			parentPkg = parentPkg.getParentPackage();
579 		}
580 
581 		if (!pkg.hasAlias()) {
582 			String pkgName = pkg.getName();
583 			if ((args.isDeobfuscationOn() && shouldRename(pkgName))
584 					|| (args.isRenameValid() && !NameMapper.isValidIdentifier(pkgName))
585 					|| (args.isRenamePrintable() && !NameMapper.isAllCharsPrintable(pkgName))) {
586 				String pkgAlias = String.format("p%03d%s", pkgIndex++, prepareNamePart(pkg.getName()));
587 				pkg.setAlias(pkgAlias);
588 			}

592 	private boolean shouldRename(String s) {

681 	private static boolean isR(ClassNode cls) {
682 		if (!cls.getClassInfo().getShortName().equals("R")) {
683 			return false;
684 		}
685 		if (!cls.getMethods().isEmpty() || !cls.getFields().isEmpty()) {
686 			return false;
687 		}
688 		for (ClassNode inner : cls.getInnerClasses()) {
689 			for (MethodNode m : inner.getMethods()) {
690 				if (!m.getMethodInfo().isConstructor() && !m.getMethodInfo().isClassInit()) {
691 					return false;
692 				}
693 			}
694 			for (FieldNode field : cls.getFields()) {
695 				ArgType type = field.getType();
696 				if (type != ArgType.INT && (!type.isArray() || type.getArrayElement() != ArgType.INT)) {
697 					return false;
698 				}
699 			}
700 		}
701 		return true;
702 	}
