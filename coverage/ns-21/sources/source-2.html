254 	private boolean checkSourceFilenameAttr() {
255 		SourceFileAttr sourceFileAttr = get(JadxAttrType.SOURCE_FILE);
256 		if (sourceFileAttr == null) {
257 			return true;
258 		}
259 		String fileName = sourceFileAttr.getFileName();
260 		if (fileName.endsWith(".java")) {
261 			fileName = fileName.substring(0, fileName.length() - 5);
262 		}
263 		if (fileName.isEmpty() || fileName.equals("SourceFile")) {
264 			return false;
265 		}
266 		if (clsInfo != null) {
267 			String name = clsInfo.getShortName();
268 			if (fileName.equals(name)) {
269 				return false;
270 			}
271 			ClassInfo parentCls = clsInfo.getParentClass();
272 			while (parentCls != null) {
273 				String parentName = parentCls.getShortName();
274 				if (parentName.equals(fileName) || parentName.startsWith(fileName + '$')) {
275 					return false;
276 				}
277 				parentCls = parentCls.getParentClass();
278 			}
279 			if (fileName.contains("$") && fileName.endsWith('$' + name)) {
280 				return false;
281 			}
282 			if (name.contains("$") && name.startsWith(fileName)) {
283 				return false;
284 			}
285 		}
286 		return true;
287 	}
