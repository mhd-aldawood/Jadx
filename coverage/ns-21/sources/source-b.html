126 	public void load() throws DecodeException {
127 		if (loaded) {
128 			// method already loaded
129 			return;
130 		}
131 		try {
132 			loaded = true;
133 			if (noCode) {
134 				regsCount = 0;
135 				// TODO: registers not needed without code
136 				initArguments(this.argTypes);
137 				return;
138 			}
139 
140 			this.regsCount = codeReader.getRegistersCount();
141 			this.argsStartReg = codeReader.getArgsStartReg();
142 			initArguments(this.argTypes);
143 			InsnDecoder decoder = new InsnDecoder(this);
144 			this.instructions = decoder.process(codeReader);
145 		} catch (Exception e) {
146 			if (!noCode) {
147 				unload();
148 				noCode = true;
149 				// load without code
150 				load();
151 				noCode = false;
152 			}
153 			throw new DecodeException(this, "Load method exception: "
154 					+ e.getClass().getSimpleName() + ": " + e.getMessage(), e);
155 		}
156 	}

167 	private void initArguments(List<ArgType> args) {
168 		int pos = getArgsStartPos(args);
169 		TypeUtils typeUtils = root().getTypeUtils();
170 		if (accFlags.isStatic()) {
171 			thisArg = null;
172 		} else {
173 			ArgType thisClsType = typeUtils.expandTypeVariables(this, parentClass.getType());
174 			RegisterArg arg = InsnArg.reg(pos++, thisClsType);
175 			arg.add(AFlag.THIS);
176 			arg.add(AFlag.IMMUTABLE_TYPE);
177 			thisArg = arg;
178 		}
179 		if (args.isEmpty()) {
180 			argsList = Collections.emptyList();
181 			return;
182 		}
183 		argsList = new ArrayList<>(args.size());
184 		for (ArgType argType : args) {
185 			ArgType expandedType = typeUtils.expandTypeVariables(this, argType);
186 			RegisterArg regArg = InsnArg.reg(pos, expandedType);
187 			regArg.add(AFlag.METHOD_ARGUMENT);
188 			regArg.add(AFlag.IMMUTABLE_TYPE);
189 			argsList.add(regArg);
190 			pos += argType.getRegCount();
191 		}
192 	}

194 	private int getArgsStartPos(List<ArgType> args) {
195 		if (noCode) {
196 			return 0;
197 		}
198 		if (argsStartReg != -1) {
199 			return argsStartReg;
200 		}
201 		int pos = regsCount;
202 		for (ArgType arg : args) {
203 			pos -= arg.getRegCount();
204 		}
205 		if (!accFlags.isStatic()) {
206 			pos--;
207 		}
208 		return pos;
209 	}
