107 		int insnsCount = classes.stream().flatMap(c -> c.getMethods().stream()).mapToInt(MethodNode::getInsnsCount).sum();

134 	private static void markDuplicatedClasses(List<ClassNode> classes) {
135 		classes.stream()
136 				.collect(Collectors.groupingBy(ClassNode::getClassInfo))
137 				.entrySet()
138 				.stream()
139 				.filter(entry -> entry.getValue().size() > 1)
140 				.forEach(entry -> {
141 					List<String> sources = Utils.collectionMap(entry.getValue(), ClassNode::getInputFileName);
142 					LOG.warn("Found duplicated class: {}, count: {}. Only one will be loaded!\n  {}",
143 							entry.getKey(), entry.getValue().size(), String.join("\n  ", sources));
144 					entry.getValue().forEach(cls -> {
145 						String thisSource = cls.getInputFileName();
146 						String otherSourceStr = sources.stream()
147 								.filter(s -> !s.equals(thisSource))
148 								.sorted()
149 								.collect(Collectors.joining("\n  "));
150 						cls.addWarnComment("Classes with same name are omitted:\n  " + otherSourceStr + '\n');
151 					});
152 				});
153 	}

236 		List<ClassNode> inner = new ArrayList<>();

239 				inner.add(cls);

241 		}

243 		for (ClassNode cls : inner) {
244 			ClassInfo clsInfo = cls.getClassInfo();
245 			ClassNode parent = resolveClass(clsInfo.getParentClass());
246 			if (parent == null) {
247 				clsMap.remove(clsInfo);
248 				clsInfo.notInner(this);
249 				clsMap.put(clsInfo, cls);
250 				updated.add(cls);
251 			} else {
252 				parent.addInnerClass(cls);
253 			}
254 		}
255 		// reload names for inner classes of updated parents
256 		for (ClassNode updCls : updated) {
257 			for (ClassNode innerCls : updCls.getInnerClasses()) {

393 	private MethodNode deepResolveMethod(@NotNull ClassNode cls, String signature) {
394 		for (MethodNode m : cls.getMethods()) {
395 			if (m.getMethodInfo().getShortId().startsWith(signature)) {
396 				return m;
397 			}
398 		}
399 		MethodNode found;
400 		ArgType superClass = cls.getSuperClass();
401 		if (superClass != null) {
402 			ClassNode superNode = resolveClass(superClass);
403 			if (superNode != null) {
404 				found = deepResolveMethod(superNode, signature);
405 				if (found != null) {
406 					return found;
407 				}
408 			}
409 		}
410 		for (ArgType iFaceType : cls.getInterfaces()) {
411 			ClassNode iFaceNode = resolveClass(iFaceType);
412 			if (iFaceNode != null) {
413 				found = deepResolveMethod(iFaceNode, signature);
414 				if (found != null) {
415 					return found;
416 				}
417 			}
418 		}
419 		return null;
420 	}
