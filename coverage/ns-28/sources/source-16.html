97 	private static void replaceStep(MethodNode mth, InsnRemover remover) {
98 		ClassNode parentClass = mth.getParentClass();
99 		for (BlockNode block : mth.getBasicBlocks()) {
100 			remover.setBlock(block);
101 			List<InsnNode> insnsList = block.getInstructions();
102 			int size = insnsList.size();
103 			for (int i = 0; i < size; i++) {
104 				InsnNode insn = insnsList.get(i);
105 				switch (insn.getType()) {
106 					case CONSTRUCTOR:
107 						processAnonymousConstructor(mth, ((ConstructorInsn) insn));
108 						break;
109 
110 					case CONST:
111 					case CONST_STR:
112 					case CONST_CLASS:
113 						replaceConst(mth, parentClass, block, i, insn);
114 						break;
115 
116 					case SWITCH:
117 						replaceConstKeys(mth, parentClass, (SwitchInsn) insn);
118 						break;
119 
120 					case NEW_ARRAY:
121 						// replace with filled array if 'fill-array' is next instruction
122 						NewArrayNode newArrInsn = (NewArrayNode) insn;
123 						InsnNode nextInsn = getFirstUseSkipMove(insn.getResult());
124 						if (nextInsn != null && nextInsn.getType() == InsnType.FILL_ARRAY) {
125 							FillArrayInsn fillArrInsn = (FillArrayInsn) nextInsn;
126 							if (checkArrSizes(mth, newArrInsn, fillArrInsn)) {
127 								InsnNode filledArr = makeFilledArrayInsn(mth, newArrInsn, fillArrInsn);
128 								replaceInsn(mth, block, i, filledArr);
129 								remover.addAndUnbind(nextInsn);
130 							}
131 						}
132 						break;
133 
134 					case MOVE_EXCEPTION:
135 						processMoveException(mth, block, insn, remover);
136 						break;
137 
138 					case ARITH:
139 						processArith(mth, parentClass, (ArithNode) insn);
140 						break;
141 
142 					case CHECK_CAST:
143 						removeCheckCast(mth, block, i, (IndexInsnNode) insn);
144 						break;
145 
146 					case CAST:
147 						fixPrimitiveCast(mth, block, i, insn);
148 						break;
149 
150 					case IPUT:
151 					case IGET:
152 						fixFieldUsage(mth, (IndexInsnNode) insn);
153 						break;
154 
155 					default:
156 						break;
157 				}
158 			}
159 			remover.perform();
160 		}

166 	private static void fixFieldUsage(MethodNode mth, IndexInsnNode insn) {
167 		InsnArg instanceArg = insn.getArg(insn.getType() == InsnType.IGET ? 0 : 1);
168 		if (instanceArg.contains(AFlag.SUPER)) {
169 			return;
170 		}
171 		if (instanceArg.isInsnWrap() && ((InsnWrapArg) instanceArg).getWrapInsn().getType() == InsnType.CAST) {
172 			return;
173 		}
174 		FieldInfo fieldInfo = (FieldInfo) insn.getIndex();
175 		ArgType clsType = fieldInfo.getDeclClass().getType();
176 		ArgType instanceType = instanceArg.getType();
177 		if (Objects.equals(clsType, instanceType)) {
178 			// cast not needed
179 			return;
180 		}
181 
182 		FieldNode fieldNode = mth.root().resolveField(fieldInfo);
183 		if (fieldNode == null) {
184 			// unknown field
185 			TypeCompareEnum result = mth.root().getTypeCompare().compareTypes(instanceType, clsType);
186 			if (result.isEqual() || (result == TypeCompareEnum.NARROW_BY_GENERIC && !instanceType.isGenericType())) {
187 				return;
188 			}
189 		} else if (isFieldVisibleInMethod(fieldNode, mth)) {
190 			return;
191 		}
192 		// insert cast
193 		IndexInsnNode castInsn = new IndexInsnNode(InsnType.CAST, clsType, 1);
194 		castInsn.addArg(instanceArg.duplicate());
195 		castInsn.add(AFlag.SYNTHETIC);
196 		castInsn.add(AFlag.EXPLICIT_CAST);
197 
198 		InsnArg castArg = InsnArg.wrapInsnIntoArg(castInsn);
199 		castArg.setType(clsType);
200 		insn.replaceArg(instanceArg, castArg);

204 	private static boolean isFieldVisibleInMethod(FieldNode field, MethodNode mth) {
205 		AccessInfo accessFlags = field.getAccessFlags();
206 		if (accessFlags.isPublic()) {
207 			return true;
208 		}
209 		ClassNode useCls = mth.getParentClass();
210 		ClassNode fieldCls = field.getParentClass();
211 		boolean sameScope = Objects.equals(useCls, fieldCls) && !mth.getAccessFlags().isStatic();
212 		if (sameScope) {
213 			return true;
214 		}
215 		if (accessFlags.isPrivate()) {
216 			return false;
217 		}
218 		// package-private or protected
219 		if (Objects.equals(useCls.getClassInfo().getPackage(), fieldCls.getClassInfo().getPackage())) {
220 			// same package
221 			return true;
222 		}
223 		if (accessFlags.isPackagePrivate()) {
224 			return false;
225 		}
226 		// protected

229 	}

445 	 * For args in anonymous constructor invoke apply:
446 	 * - forbid inline into constructor call
447 	 * - make variables final (compiler require this implicitly)
448 	 */
449 	private static void processAnonymousConstructor(MethodNode mth, ConstructorInsn co) {
450 		IMethodDetails callMthDetails = mth.root().getMethodUtils().getMethodDetails(co);
451 		if (!(callMthDetails instanceof MethodNode)) {
452 			return;
453 		}
454 		MethodNode callMth = (MethodNode) callMthDetails;
455 		if (!callMth.contains(AFlag.ANONYMOUS_CONSTRUCTOR) || callMth.contains(AFlag.NO_SKIP_ARGS)) {
456 			return;
457 		}
458 		SkipMethodArgsAttr attr = callMth.get(AType.SKIP_MTH_ARGS);
459 		if (attr != null) {
460 			int argsCount = Math.min(callMth.getArgRegs().size(), co.getArgsCount());
461 			for (int i = 0; i < argsCount; i++) {
462 				if (attr.isSkip(i)) {
463 					anonymousCallArgMod(co.getArg(i));
464 				}
465 			}
466 		} else {
467 			// additional info not available apply mods to all args (the safest solution)
468 			co.getArguments().forEach(ModVisitor::anonymousCallArgMod);
469 		}
470 	}

501 	private static InsnNode makeFilledArrayInsn(MethodNode mth, NewArrayNode newArrayNode, FillArrayInsn insn) {
502 		ArgType insnArrayType = newArrayNode.getArrayType();
503 		ArgType insnElementType = insnArrayType.getArrayElement();
504 		ArgType elType = insn.getElementType();
505 		if (!elType.isTypeKnown()
506 				&& insnElementType.isPrimitive()
507 				&& elType.contains(insnElementType.getPrimitiveType())) {
508 			elType = insnElementType;
509 		}
510 		if (!elType.equals(insnElementType) && !insnArrayType.equals(ArgType.OBJECT)) {
511 			mth.addWarn("Incorrect type for fill-array insn " + InsnUtils.formatOffset(insn.getOffset())
512 					+ ", element type: " + elType + ", insn element type: " + insnElementType);
513 		}
514 		if (!elType.isTypeKnown()) {
515 			LOG.warn("Unknown array element type: {} in mth: {}", elType, mth);
516 			elType = insnElementType.isTypeKnown() ? insnElementType : elType.selectFirst();
517 			if (elType == null) {
518 				throw new JadxRuntimeException("Null array element type");
519 			}
520 		}
521 
522 		List<LiteralArg> list = insn.getLiteralArgs(elType);
523 		InsnNode filledArr = new FilledNewArrayNode(elType, list.size());
524 		filledArr.setResult(newArrayNode.getResult());
525 		for (LiteralArg arg : list) {
526 			FieldNode f = mth.getParentClass().getConstFieldByLiteralArg(arg);
527 			if (f != null) {
528 				InsnNode fGet = new IndexInsnNode(InsnType.SGET, f.getFieldInfo(), 0);
