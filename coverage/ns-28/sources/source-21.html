324 	private static List<InsnNode> collectUseChain(MethodNode mth, InvokeNode insn, RegisterArg instanceArg) {
325 		SSAVar sVar = instanceArg.getSVar();
326 		if (sVar.isUsedInPhi() || sVar.getUseCount() == 0) {
327 			return Collections.emptyList();
328 		}
329 		List<InsnNode> useChain = new ArrayList<>(sVar.getUseCount() + 1);
330 		InsnNode assignInsn = sVar.getAssign().getParentInsn();
331 		if (assignInsn == null) {
332 			return Collections.emptyList();
333 		}
334 		useChain.add(assignInsn);
335 		for (RegisterArg reg : sVar.getUseList()) {
336 			InsnNode parentInsn = reg.getParentInsn();
337 			if (parentInsn == null) {
338 				return Collections.emptyList();
339 			}
340 			useChain.add(parentInsn);
341 		}
342 		int toStrIdx = InsnList.getIndex(useChain, insn);
343 		if (useChain.size() - 1 != toStrIdx) {
344 			return Collections.emptyList();
345 		}
346 		useChain.remove(toStrIdx);
347 
348 		// all insns must be in one block and sequential
349 		BlockNode assignBlock = BlockUtils.getBlockByInsn(mth, assignInsn);
350 		if (assignBlock == null) {
351 			return Collections.emptyList();
352 		}
353 		List<InsnNode> blockInsns = assignBlock.getInstructions();
354 		int assignIdx = InsnList.getIndex(blockInsns, assignInsn);
355 		int chainSize = useChain.size();
356 		int lastInsn = blockInsns.size() - assignIdx;
357 		if (lastInsn < chainSize) {
358 			return Collections.emptyList();
359 		}
360 		for (int i = 1; i < chainSize; i++) {
361 			if (blockInsns.get(assignIdx + i) != useChain.get(i)) {
362 				return Collections.emptyList();
363 			}
364 		}
365 		return useChain;
366 	}

517 		List<InsnNode> chain = new ArrayList<>();
518 		InsnArg arg = insn.getArg(0);

543 	private static InsnNode simplifyArith(ArithNode arith) {
544 		if (arith.getArgsCount() != 2) {
545 			return null;
546 		}
547 		LiteralArg litArg = null;
548 		InsnArg secondArg = arith.getArg(1);
549 		if (secondArg.isInsnWrap()) {
550 			InsnNode wr = ((InsnWrapArg) secondArg).getWrapInsn();
551 			if (wr.getType() == InsnType.CONST) {
552 				InsnArg arg = wr.getArg(0);
553 				if (arg.isLiteral()) {
554 					litArg = (LiteralArg) arg;
555 				}
556 			}
557 		} else if (secondArg.isLiteral()) {
558 			litArg = (LiteralArg) secondArg;
559 		}
560 		if (litArg == null) {
561 			return null;
562 		}
563 		switch (arith.getOp()) {
564 			case ADD:
565 				// fix 'c + (-1)' to 'c - (1)'
566 				if (litArg.isNegative()) {
567 					LiteralArg negLitArg = litArg.negate();
568 					if (negLitArg != null) {
569 						return new ArithNode(ArithOp.SUB, arith.getResult(), arith.getArg(0), negLitArg);
570 					}
571 				}
572 				break;
573 
574 			case XOR:
575 				// simplify xor on boolean
576 				InsnArg firstArg = arith.getArg(0);
577 				long lit = litArg.getLiteral();
578 				if (firstArg.getType() == ArgType.BOOLEAN && (lit == 0 || lit == 1)) {
579 					InsnNode node = new InsnNode(lit == 0 ? InsnType.MOVE : InsnType.NOT, 1);
580 					node.setResult(arith.getResult());
581 					node.addArg(firstArg);
582 					return node;
583 				}
584 				break;
585 		}
586 		return null;
587 	}

589 	/**
590 	 * Convert field arith operation to arith instruction
591 	 * (IPUT (ARITH (IGET, lit)) -> ARITH ((IGET)) <op>= lit))
592 	 */
593 	private static ArithNode convertFieldArith(MethodNode mth, InsnNode insn) {
594 		InsnArg arg = insn.getArg(0);
595 		if (!arg.isInsnWrap()) {
596 			return null;
597 		}
598 		InsnNode wrap = ((InsnWrapArg) arg).getWrapInsn();
599 		InsnType wrapType = wrap.getType();
600 		if (wrapType != InsnType.ARITH && wrapType != InsnType.STR_CONCAT
601 				|| !wrap.getArg(0).isInsnWrap()) {
602 			return null;
603 		}
604 		InsnArg getWrap = wrap.getArg(0);
605 		InsnNode get = ((InsnWrapArg) getWrap).getWrapInsn();
606 		InsnType getType = get.getType();
607 		if (getType != InsnType.IGET && getType != InsnType.SGET) {
608 			return null;
609 		}
610 		FieldInfo field = (FieldInfo) ((IndexInsnNode) insn).getIndex();
611 		FieldInfo innerField = (FieldInfo) ((IndexInsnNode) get).getIndex();
612 		if (!field.equals(innerField)) {
613 			return null;
614 		}
615 		try {
616 			if (getType == InsnType.IGET && insn.getType() == InsnType.IPUT) {
617 				InsnArg reg = get.getArg(0);
618 				InsnArg putReg = insn.getArg(1);
619 				if (!reg.equals(putReg)) {
620 					return null;
621 				}
622 			}
623 			InsnArg fArg = getWrap.duplicate();
624 			InsnRemover.unbindInsn(mth, get);
625 			if (insn.getType() == InsnType.IPUT) {
626 				InsnRemover.unbindArgUsage(mth, insn.getArg(1));
627 			}
628 			if (wrapType == InsnType.ARITH) {
629 				ArithNode ar = (ArithNode) wrap;
630 				return ArithNode.oneArgOp(ar.getOp(), fArg, ar.getArg(1));
631 			}
632 			int argsCount = wrap.getArgsCount();
633 			InsnNode concat = new InsnNode(InsnType.STR_CONCAT, argsCount - 1);
634 			for (int i = 1; i < argsCount; i++) {
635 				concat.addArg(wrap.getArg(i));
636 			}
637 			return ArithNode.oneArgOp(ArithOp.ADD, fArg, InsnArg.wrapArg(concat));
638 		} catch (Exception e) {
639 			LOG.debug("Can't convert field arith insn: {}, mth: {}", insn, mth, e);
640 		}
641 		return null;
642 	}
