91 	private boolean convertToEnum(ClassNode cls) {
92 		if (!cls.isEnum()) {
93 			return false;
94 		}
95 		MethodNode classInitMth = cls.getClassInitMth();
96 		if (classInitMth == null) {
97 			cls.addWarnComment("Enum class init method not found");
98 			return false;
99 		}
100 		if (classInitMth.getBasicBlocks().isEmpty()) {
101 			return false;
102 		}
103 		ArgType clsType = cls.getClassInfo().getType();
104 
105 		// search "$VALUES" field (holds all enum values)
106 		List<FieldNode> valuesCandidates = cls.getFields().stream()
107 				.filter(f -> f.getAccessFlags().isStatic())
108 				.filter(f -> f.getType().isArray())
109 				.filter(f -> Objects.equals(f.getType().getArrayRootElement(), clsType))
110 				.collect(Collectors.toList());
111 
112 		if (valuesCandidates.isEmpty()) {
113 			return false;
114 		}
115 		if (valuesCandidates.size() > 1) {
116 			valuesCandidates.removeIf(f -> !f.getAccessFlags().isSynthetic());
117 		}
118 		if (valuesCandidates.size() > 1) {
119 			Optional<FieldNode> valuesOpt = valuesCandidates.stream().filter(f -> f.getName().equals("$VALUES")).findAny();
120 			if (valuesOpt.isPresent()) {
121 				valuesCandidates.clear();
122 				valuesCandidates.add(valuesOpt.get());
123 			}
124 		}
125 		if (valuesCandidates.size() != 1) {
126 			cls.addWarnComment("Found several \"values\" enum fields: " + valuesCandidates);
127 			return false;
128 		}
129 		FieldNode valuesField = valuesCandidates.get(0);
130 		List<InsnNode> toRemove = new ArrayList<>();
131 
132 		// search "$VALUES" array init and collect enum fields
133 		BlockInsnPair valuesInitPair = getValuesInitInsn(classInitMth, valuesField);
134 		if (valuesInitPair == null) {
135 			return false;
136 		}
137 		BlockNode staticBlock = valuesInitPair.getBlock();
138 		InsnNode valuesInitInsn = valuesInitPair.getInsn();
139 
140 		List<EnumField> enumFields = null;
141 		InsnArg arrArg = valuesInitInsn.getArg(0);
142 		if (arrArg.isInsnWrap()) {
143 			InsnNode wrappedInsn = ((InsnWrapArg) arrArg).getWrapInsn();
144 			enumFields = extractEnumFieldsFromInsn(cls, staticBlock, wrappedInsn, toRemove);
145 		}
146 		if (enumFields == null) {
147 			return false;
148 		}
149 		toRemove.add(valuesInitInsn);
150 
151 		// all checks complete, perform transform
152 		EnumClassAttr attr = new EnumClassAttr(enumFields);
153 		attr.setStaticMethod(classInitMth);
154 		cls.addAttr(attr);
155 
156 		for (EnumField enumField : attr.getFields()) {
157 			ConstructorInsn co = enumField.getConstrInsn();
158 			FieldNode fieldNode = enumField.getField();
159 
160 			// use string arg from the constructor as enum field name
161 			String name = getConstString(cls.root(), co.getArg(0));
162 			if (name != null
163 					&& !fieldNode.getAlias().equals(name)
164 					&& NameMapper.isValidAndPrintable(name)
165 					&& cls.root().getArgs().isRenameValid()) {
166 				fieldNode.getFieldInfo().setAlias(name);
167 			}
168 			fieldNode.add(AFlag.DONT_GENERATE);
169 			processConstructorInsn(cls, enumField, classInitMth, staticBlock, toRemove);
170 		}
171 		valuesField.add(AFlag.DONT_GENERATE);
172 		InsnRemover.removeAllAndUnbind(classInitMth, staticBlock, toRemove);
173 		if (classInitMth.countInsns() == 0) {
174 			classInitMth.add(AFlag.DONT_GENERATE);
175 		} else if (!toRemove.isEmpty()) {
176 			CodeShrinkVisitor.shrinkMethod(classInitMth);
177 		}
178 		removeEnumMethods(cls, clsType, valuesField);
179 		return true;
180 	}

182 	private void processConstructorInsn(ClassNode cls, EnumField enumField, MethodNode classInitMth,
183 			BlockNode staticBlock, List<InsnNode> toRemove) {
184 		ConstructorInsn co = enumField.getConstrInsn();
185 		ClassInfo enumClsInfo = co.getClassType();
186 		if (!enumClsInfo.equals(cls.getClassInfo())) {
187 			ClassNode enumCls = cls.root().resolveClass(enumClsInfo);
188 			if (enumCls != null) {
189 				processEnumCls(cls, enumField, enumCls);
190 			}
191 		}
192 		List<RegisterArg> regs = new ArrayList<>();
193 		co.getRegisterArgs(regs);
194 		if (!regs.isEmpty()) {
195 			cls.addWarnComment("Init of enum " + enumField.getField().getName() + " can be incorrect");
196 		}
197 		MethodNode ctrMth = cls.root().resolveMethod(co.getCallMth());
198 		if (ctrMth != null) {
199 			markArgsForSkip(ctrMth);
200 		}
201 		RegisterArg coResArg = co.getResult();
202 		if (coResArg == null || coResArg.getSVar().getUseList().size() <= 2) {
203 			toRemove.add(co);
204 		} else {
205 			// constructor result used in other places -> replace constructor with enum field get (SGET)
206 			IndexInsnNode enumGet = new IndexInsnNode(InsnType.SGET, enumField.getField().getFieldInfo(), 0);
207 			enumGet.setResult(coResArg.duplicate());
208 			BlockUtils.replaceInsn(classInitMth, staticBlock, co, enumGet);
209 		}
210 	}

213 	private List<EnumField> extractEnumFieldsFromInsn(ClassNode cls, BlockNode staticBlock,
214 			InsnNode wrappedInsn, List<InsnNode> toRemove) {
215 		switch (wrappedInsn.getType()) {
216 			case FILLED_NEW_ARRAY:
217 				return extractEnumFieldsFromFilledArray(cls, wrappedInsn, staticBlock, toRemove);
218 
219 			case INVOKE:
220 				// handle redirection of values array fill (added in java 15)
221 				return extractEnumFieldsFromInvoke(cls, staticBlock, (InvokeNode) wrappedInsn, toRemove);
222 
223 			case NEW_ARRAY:
224 				InsnArg arg = wrappedInsn.getArg(0);
225 				if (arg.isZeroLiteral()) {
226 					// empty enum
227 					return Collections.emptyList();
228 				}
229 				return null;
230 
231 			default:
232 				return null;
233 		}
234 	}

237 			InvokeNode invokeNode, List<InsnNode> toRemove) {
238 		MethodInfo callMth = invokeNode.getCallMth();
239 		MethodNode valuesMth = cls.root().resolveMethod(callMth);
240 		if (valuesMth == null || valuesMth.isVoidReturn()) {
241 			return null;
242 		}
243 		BlockNode returnBlock = Utils.getOne(valuesMth.getPreExitBlocks());
244 		InsnNode returnInsn = BlockUtils.getLastInsn(returnBlock);
245 		InsnNode wrappedInsn = getWrappedInsn(getSingleArg(returnInsn));
246 		if (wrappedInsn == null) {
247 			return null;
248 		}
249 		List<EnumField> enumFields = extractEnumFieldsFromInsn(cls, staticBlock, wrappedInsn, toRemove);
250 		if (enumFields != null) {
251 			valuesMth.add(AFlag.DONT_GENERATE);
252 		}
253 		return enumFields;
254 	}

280 			} else if (arg.isRegister()) {

282 			}
283 			if (field == null) {
284 				return null;

287 		}

293 		if (wrappedInsn.getType() == InsnType.SGET) {
294 			return processEnumFieldByField(cls, wrappedInsn, staticBlock, toRemove);
295 		}
296 		ConstructorInsn constructorInsn = castConstructorInsn(wrappedInsn);
297 		if (constructorInsn != null) {
298 			FieldNode enumFieldNode = createFakeField(cls, "EF" + constructorInsn.getOffset());
299 			cls.addField(enumFieldNode);
300 			return createEnumFieldByConstructor(cls, enumFieldNode, constructorInsn);
301 		}
302 		return null;

307 		if (sgetInsn.getType() != InsnType.SGET) {
308 			return null;
309 		}

311 		FieldNode enumFieldNode = cls.searchField(fieldInfo);
312 		if (enumFieldNode == null) {

314 		}

316 		if (sputInsn == null) {

321 		if (co == null) {
322 			return null;
323 		}

325 		if (sgetResult == null || sgetResult.getSVar().getUseCount() == 1) {

334 		InsnNode assignInsn = arg.getAssignInsn();
335 		if (assignInsn != null && assignInsn.getType() == InsnType.SGET) {
336 			return processEnumFieldByField(cls, assignInsn, staticBlock, toRemove);
337 		}
338

340 		if (ssaVar.getUseCount() == 0) {
341 			return null;
342 		}
343 		InsnNode constrInsn = ssaVar.getAssign().getParentInsn();
344 		if (constrInsn == null || constrInsn.getType() != InsnType.CONSTRUCTOR) {
345 			return null;
346 		}
347 		FieldNode enumFieldNode = searchEnumField(cls, ssaVar, toRemove);
348 		if (enumFieldNode == null) {
349 			enumFieldNode = createFakeField(cls, "EF" + arg.getRegNum());
350 			cls.addField(enumFieldNode);
351 		}
352 		return createEnumFieldByConstructor(cls, enumFieldNode, (ConstructorInsn) constrInsn);

379 	private EnumField createEnumFieldByConstructor(ClassNode cls, FieldNode enumFieldNode, ConstructorInsn co) {
380 		// usually constructor signature is '<init>(Ljava/lang/String;I)V'.
381 		// sometimes for one field enum second arg can be omitted
382 		if (co.getArgsCount() < 1) {
383 			return null;
384 		}
385 		ClassInfo clsInfo = co.getClassType();
386 		ClassNode constrCls = cls.root().resolveClass(clsInfo);
387 		if (constrCls == null) {
388 			return null;
389 		}
390 		if (constrCls.equals(cls)) {
391 			// allow same class
392 		} else if (constrCls.contains(AFlag.ANONYMOUS_CLASS)) {
393 			// allow external class already marked as anonymous
394 		} else {
395 			return null;
396 		}
397 		MethodNode ctrMth = cls.root().resolveMethod(co.getCallMth());
398 		if (ctrMth == null) {
399 			return null;
400 		}
401 		return new EnumField(enumFieldNode, co);

418 	private void removeEnumMethods(ClassNode cls, ArgType clsType, FieldNode valuesField) {
419 		String valuesMethod = "values()" + TypeGen.signature(ArgType.array(clsType));
420 		FieldInfo valuesFieldInfo = valuesField.getFieldInfo();
421 
422 		// remove compiler generated methods
423 		for (MethodNode mth : cls.getMethods()) {
424 			MethodInfo mi = mth.getMethodInfo();
425 			if (mi.isClassInit()) {
426 				continue;
427 			}
428 			String shortId = mi.getShortId();
429 			if (mi.isConstructor()) {
430 				if (isDefaultConstructor(mth, shortId)) {
431 					mth.add(AFlag.DONT_GENERATE);
432 				}
433 				markArgsForSkip(mth);
434 			} else if (shortId.equals(valuesMethod)
435 					|| usesValuesField(mth, valuesFieldInfo)
436 					|| simpleValueOfMth(mth, clsType)) {
437 				mth.add(AFlag.DONT_GENERATE);
438 			}
439 		}
440 	}

450 	private boolean isDefaultConstructor(MethodNode mth, String shortId) {
451 		boolean defaultId = shortId.equals("<init>(Ljava/lang/String;I)V")
452 				|| shortId.equals("<init>(Ljava/lang/String;)V");
453 		if (defaultId) {
454 			// check content
455 			return mth.countInsns() == 0;
456 		}

460 	private boolean simpleValueOfMth(MethodNode mth, ArgType clsType) {
461 		InsnNode returnInsn = InsnUtils.searchSingleReturnInsn(mth, insn -> insn.getArgsCount() == 1);
462 		if (returnInsn == null) {
463 			return false;
464 		}
465 		InsnNode wrappedInsn = getWrappedInsn(getSingleArg(returnInsn));
466 		IndexInsnNode castInsn = (IndexInsnNode) checkInsnType(wrappedInsn, InsnType.CHECK_CAST);
467 		if (castInsn != null && Objects.equals(castInsn.getIndex(), clsType)) {
468 			InvokeNode invokeInsn = (InvokeNode) checkInsnType(getWrappedInsn(getSingleArg(castInsn)), InsnType.INVOKE);
469 			return invokeInsn != null && invokeInsn.getCallMth().equals(enumValueOfMth);
470 		}
471 		return false;
472 	}

480 		// remove constructor, because it is anonymous class
481 		for (MethodNode innerMth : innerCls.getMethods()) {

488 			// not inner

495 		if (insn.getArgsCount() != 1) {
496 			return null;
497 		}

499 		if (arg.isInsnWrap()) {

501 		}

503 			return castConstructorInsn(((RegisterArg) arg).getAssignInsn());
504 		}
