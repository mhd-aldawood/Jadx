72 	 * Sometimes try block can be handler block itself and should be connected before wrapping.
73 	 * Use queue for postpone try blocks not ready for wrap.
74 	 */
75 	private static void connectExcHandlers(MethodNode mth, List<TryCatchBlockAttr> tryBlocks) {
76 		if (tryBlocks.isEmpty()) {
77 			return;
78 		}
79 		int limit = tryBlocks.size() * 3;
80 		int count = 0;
81 		Deque<TryCatchBlockAttr> queue = new ArrayDeque<>(tryBlocks);
82 		while (!queue.isEmpty()) {
83 			TryCatchBlockAttr tryBlock = queue.removeFirst();
84 			boolean complete = wrapBlocksWithTryCatch(mth, tryBlock);
85 			if (!complete) {
86 				queue.addLast(tryBlock); // return to queue at the end
87 			}
88 			if (count++ > limit) {
89 				throw new JadxRuntimeException("Try blocks wrapping queue limit reached! Please report as an issue!");
90 			}
91 		}

264 	private static boolean checkTryCatchRelation(List<TryCatchBlockAttr> tryBlocks,
265 			TryCatchBlockAttr outerTryBlock, TryCatchBlockAttr innerTryBlock) {
266 		if (outerTryBlock.getBlocks().equals(innerTryBlock.getBlocks())) {
267 			// same try blocks -> merge handlers
268 			List<ExceptionHandler> handlers = Utils.concatDistinct(outerTryBlock.getHandlers(), innerTryBlock.getHandlers());
269 			tryBlocks.add(new TryCatchBlockAttr(tryBlocks.size(), handlers, outerTryBlock.getBlocks()));
270 			tryBlocks.remove(outerTryBlock);
271 			tryBlocks.remove(innerTryBlock);
272 			return true;
273 		}
274 
275 		Set<BlockNode> handlerBlocks = innerTryBlock.getHandlers().stream()
276 				.flatMap(eh -> eh.getBlocks().stream())
277 				.collect(Collectors.toSet());
278 		boolean catchInHandler = handlerBlocks.stream().anyMatch(isHandlersIntersects(outerTryBlock));
279 		boolean catchInTry = innerTryBlock.getBlocks().stream().anyMatch(isHandlersIntersects(outerTryBlock));
280 		boolean blocksOutsideHandler = outerTryBlock.getBlocks().stream().anyMatch(b -> !handlerBlocks.contains(b));
281 
282 		boolean makeInner = catchInHandler && (catchInTry || blocksOutsideHandler);
283 		if (makeInner && innerTryBlock.isAllHandler()) {
284 			// inner try block can't have catch-all handler
285 			outerTryBlock.setBlocks(Utils.concatDistinct(outerTryBlock.getBlocks(), innerTryBlock.getBlocks()));
286 			innerTryBlock.clear();
287 			return false;
288 		}
289 		if (makeInner) {
290 			// convert to inner
291 			List<BlockNode> mergedBlocks = Utils.concatDistinct(outerTryBlock.getBlocks(), innerTryBlock.getBlocks());
292 			innerTryBlock.getHandlers().removeAll(outerTryBlock.getHandlers());
293 			innerTryBlock.setOuterTryBlock(outerTryBlock);
294 			outerTryBlock.addInnerTryBlock(innerTryBlock);
295 			outerTryBlock.setBlocks(mergedBlocks);
296 			return false;
297 		}
298 		if (innerTryBlock.getHandlers().containsAll(outerTryBlock.getHandlers())) {
299 			// merge
300 			List<BlockNode> mergedBlocks = Utils.concatDistinct(outerTryBlock.getBlocks(), innerTryBlock.getBlocks());
301 			List<ExceptionHandler> handlers = Utils.concatDistinct(outerTryBlock.getHandlers(), innerTryBlock.getHandlers());
302 			tryBlocks.add(new TryCatchBlockAttr(tryBlocks.size(), handlers, mergedBlocks));
303 			tryBlocks.remove(outerTryBlock);
304 			tryBlocks.remove(innerTryBlock);
305 			return true;
306 		}
307 		return false;
308 	}

344 			bottomSplitterBlock = existBottomSplitter != null ? existBottomSplitter : BlockSplitter.startNewBlock(mth, -1);
345 			bottomSplitterBlock.add(AFlag.EXC_BOTTOM_SPLITTER);

352 					tryCatchBlock.id(), topSplitterBlock, bottomSplitterBlock);

354 		connectSplittersAndHandlers(tryCatchBlock, topSplitterBlock, bottomSplitterBlock);

358 			if (currentTCBAttr == null || currentTCBAttr.getInnerTryBlocks().contains(tryCatchBlock)) {
359 				block.addAttr(tryCatchBlock);
360 			}
361 		}
362 		tryCatchBlock.setTopSplitter(topSplitterBlock);
363

365 		if (bottomSplitterBlock != null) {
366 			bottomSplitterBlock.updateCleanSuccessors();
367 		}

369 	}

412 		// search common post-dominator block inside input set
413 		BlockNode bottom = BlockUtils.getBottomBlock(blocks);
414 		if (bottom != null) {
415 			return bottom;
416 		}
417 		// not found -> blocks don't have same dominator
418 		// try to search common cross block outside input set
419 		// NOTE: bottom block not needed for exit nodes (no data flow from them)
420 		BlockNode pathCross = BlockUtils.getPathCross(mth, blocks);
421 		if (pathCross == null) {
422 			return null;
423 		}
424 		List<BlockNode> preds = new ArrayList<>(pathCross.getPredecessors());
425 		preds.removeAll(blocks);
426 		List<BlockNode> outsidePredecessors = preds.stream()
427 				.filter(p -> !BlockUtils.atLeastOnePathExists(blocks, p))
428 				.collect(Collectors.toList());
429 		if (outsidePredecessors.isEmpty()) {
430 			return pathCross;
431 		}
432 		// some predecessors outside of input set paths -> split block only for input set
433 		BlockNode splitCross = BlockSplitter.blockSplitTop(mth, pathCross);
434 		splitCross.add(AFlag.SYNTHETIC);
435 		for (BlockNode outsidePredecessor : outsidePredecessors) {
436 			// return predecessors to split bottom block (original)
437 			BlockSplitter.replaceConnection(outsidePredecessor, splitCross, pathCross);
438 		}
439 		return splitCross;
440 	}

504 	private static boolean mergeMultiCatch(MethodNode mth, TryCatchBlockAttr tryCatch) {
505 		if (tryCatch.getHandlers().size() < 2) {
506 			return false;
507 		}
508 		for (ExceptionHandler handler : tryCatch.getHandlers()) {
509 			if (handler.getBlocks().size() != 1) {
510 				return false;
511 			}
512 			BlockNode block = handler.getHandlerBlock();
513 			if (block.getInstructions().size() != 1
514 					|| !BlockUtils.checkLastInsnType(block, InsnType.MOVE_EXCEPTION)) {
515 				return false;
516 			}
517 		}
518 		List<BlockNode> handlerBlocks = ListUtils.map(tryCatch.getHandlers(), ExceptionHandler::getHandlerBlock);
519 		List<BlockNode> successorBlocks = handlerBlocks.stream()
520 				.flatMap(h -> h.getSuccessors().stream())
521 				.distinct()
522 				.collect(Collectors.toList());
523 		if (successorBlocks.size() != 1) {
524 			return false;
525 		}
526 		BlockNode successorBlock = successorBlocks.get(0);
527 		if (!ListUtils.unorderedEquals(successorBlock.getPredecessors(), handlerBlocks)) {
528 			return false;
529 		}
530 		List<RegisterArg> regs = tryCatch.getHandlers().stream()
531 				.map(h -> Objects.requireNonNull(BlockUtils.getLastInsn(h.getHandlerBlock())).getResult())
532 				.distinct()
533 				.collect(Collectors.toList());
534 		if (regs.size() != 1) {
535 			return false;
536 		}
537 
538 		// merge confirm, leave only first handler, remove others
539 		ExceptionHandler resultHandler = tryCatch.getHandlers().get(0);
540 		tryCatch.getHandlers().removeIf(handler -> {
541 			if (handler == resultHandler) {
542 				return false;
543 			}
544 			resultHandler.addCatchTypes(handler.getCatchTypes());
545 			handler.markForRemove();
546 			return true;
547 		});
548 		return true;
549 	}

551 	private static void sortHandlers(MethodNode mth, List<TryCatchBlockAttr> tryBlocks) {
552 		TypeCompare typeCompare = mth.root().getTypeCompare();
553 		Comparator<ArgType> comparator = typeCompare.getReversedComparator();
554 		for (TryCatchBlockAttr tryBlock : tryBlocks) {
555 			for (ExceptionHandler handler : tryBlock.getHandlers()) {
556 				handler.getCatchTypes().sort((first, second) -> compareByTypeAndName(comparator, first, second));
557 			}
558 			tryBlock.getHandlers().sort((first, second) -> {
559 				if (first.equals(second)) {
560 					throw new JadxRuntimeException("Same handlers in try block: " + tryBlock);
561 				}
562 				if (first.isCatchAll()) {
563 					return 1;
564 				}
565 				if (second.isCatchAll()) {
566 					return -1;
567 				}
568 				return compareByTypeAndName(comparator,
569 						ListUtils.first(first.getCatchTypes()), ListUtils.first(second.getCatchTypes()));
570 			});
571 		}
572 	}

575 	private static int compareByTypeAndName(Comparator<ArgType> comparator, ClassInfo first, ClassInfo second) {
576 		int r = comparator.compare(first.getType(), second.getType());
577 		if (r == -2) {
578 			// on conflict sort by name
579 			return first.compareTo(second);
580 		}
581 		return r;
