230 	private static void calcDominators(List<BlockNode> basicBlocks, BlockNode entryBlock) {
231 		entryBlock.getDoms().clear();
232 		entryBlock.getDoms().set(entryBlock.getId());
233 
234 		BitSet domSet = new BitSet(basicBlocks.size());
235 		boolean changed;
236 		do {
237 			changed = false;
238 			for (BlockNode block : basicBlocks) {
239 				if (block == entryBlock) {
240 					continue;
241 				}
242 				BitSet d = block.getDoms();
243 				if (!changed) {
244 					domSet.clear();
245 					domSet.or(d);
246 				}
247 				for (BlockNode pred : block.getPredecessors()) {
248 					d.and(pred.getDoms());
249 				}
250 				d.set(block.getId());
251 				if (!changed && !d.equals(domSet)) {

253 				}
254 			}

258 	private static void calcImmediateDominators(MethodNode mth, List<BlockNode> basicBlocks, BlockNode entryBlock) {
259 		for (BlockNode block : basicBlocks) {
260 			if (block == entryBlock) {
261 				continue;
262 			}
263 			BlockNode idom;
264 			List<BlockNode> preds = block.getPredecessors();
265 			if (preds.size() == 1) {
266 				idom = preds.get(0);
267 			} else {
268 				BitSet bs = new BitSet(block.getDoms().length());
269 				bs.or(block.getDoms());
270 				for (int i = bs.nextSetBit(0); i >= 0; i = bs.nextSetBit(i + 1)) {
271 					BlockNode dom = basicBlocks.get(i);
272 					bs.andNot(dom.getDoms());
273 				}
274 				if (bs.cardinality() != 1) {
275 					throw new JadxRuntimeException("Can't find immediate dominator for block " + block
276 							+ " in " + bs + " preds:" + preds);
277 				}
278 				idom = basicBlocks.get(bs.nextSetBit(0));
279 			}
280 			block.setIDom(idom);
281 			idom.addDominatesOn(block);
282 		}
283 	}

286 		mth.getExitBlock().setDomFrontier(EMPTY);

289 		stack.push(mth.getEnterBlock());

294 			}

296 		}

298 		for (BlockNode block : domSortedBlocks) {
299 			try {
300 				computeBlockDF(mth, block);
301 			} catch (Exception e) {

303 			}

307 	private static void computeBlockDF(MethodNode mth, BlockNode block) {
308 		if (block.getDomFrontier() != null) {
309 			return;
310 		}
311 		List<BlockNode> blocks = mth.getBasicBlocks();
312 		BitSet domFrontier = null;
313 		for (BlockNode s : block.getSuccessors()) {
314 			if (s.getIDom() != block) {
315 				if (domFrontier == null) {
316 					domFrontier = new BitSet(blocks.size());
317 				}
318 				domFrontier.set(s.getId());
319 			}
320 		}
321 		for (BlockNode c : block.getDominatesOn()) {
322 			BitSet frontier = c.getDomFrontier();
323 			if (frontier == null) {
324 				throw new JadxRuntimeException("Dominance frontier not calculated for dominated block: " + c + ", from: " + block);
325 			}
326 			for (int p = frontier.nextSetBit(0); p >= 0; p = frontier.nextSetBit(p + 1)) {
327 				if (blocks.get(p).getIDom() != block) {
328 					if (domFrontier == null) {
329 						domFrontier = new BitSet(blocks.size());
330 					}
331 					domFrontier.set(p);
332 				}
333 			}
334 		}
335 		if (domFrontier == null || domFrontier.isEmpty()) {
336 			domFrontier = EMPTY;
337 		}
338 		block.setDomFrontier(domFrontier);
339 	}

467 		if (!block.contains(AFlag.LOOP_START)) {
468 			return false;
469 		}

471 		int loopsCount = loops.size();

473 			return false;
474 		}

477 				return true;
478 			}
479 		}

485 			return insertBlocksForContinue(mth, loop)

587 			// several back edges connected to one loop header => make additional block
588 			BlockNode newLoopEnd = BlockSplitter.startNewBlock(mth, block.getStartOffset());
589 			newLoopEnd.add(AFlag.SYNTHETIC);
590 			connect(newLoopEnd, block);
591 			for (LoopInfo la : loops) {
592 				BlockSplitter.replaceConnection(la.getEnd(), block, newLoopEnd);
593 			}
594 			return true;
595 		}
596 		return false;
597 	}

624 	/**
625 	 * Splice return block if several predecessors presents
626 	 */
627 	private static boolean splitReturn(MethodNode mth, BlockNode returnBlock) {
628 		if (returnBlock.contains(AFlag.SYNTHETIC)
629 				|| returnBlock.contains(AFlag.ORIG_RETURN)
630 				|| returnBlock.contains(AType.EXC_HANDLER)) {
631 			return false;
632 		}
633 		List<BlockNode> preds = returnBlock.getPredecessors();
634 		if (preds.size() < 2) {
635 			return false;
636 		}
637 		InsnNode returnInsn = BlockUtils.getLastInsn(returnBlock);
638 		if (returnInsn == null) {
639 			return false;
640 		}
641 		if (returnInsn.getArgsCount() == 1
642 				&& returnBlock.getInstructions().size() == 1
643 				&& !isReturnArgAssignInPred(preds, returnInsn)) {
644 			return false;
645 		}
646 
647 		boolean first = true;
648 		for (BlockNode pred : new ArrayList<>(preds)) {
649 			if (first) {
650 				returnBlock.add(AFlag.ORIG_RETURN);
651 				first = false;
652 			} else {
653 				BlockNode newRetBlock = BlockSplitter.startNewBlock(mth, -1);
654 				newRetBlock.add(AFlag.SYNTHETIC);
655 				newRetBlock.add(AFlag.RETURN);
656 				for (InsnNode oldInsn : returnBlock.getInstructions()) {
657 					InsnNode copyInsn = oldInsn.copyWithoutSsa();
658 					copyInsn.add(AFlag.SYNTHETIC);
659 					newRetBlock.getInstructions().add(copyInsn);
660 				}
661 				BlockSplitter.replaceConnection(pred, returnBlock, newRetBlock);
662 			}
663 		}
664 		return true;
665 	}

667 	private static boolean isReturnArgAssignInPred(List<BlockNode> preds, InsnNode returnInsn) {
668 		InsnArg retArg = returnInsn.getArg(0);
669 		if (retArg.isRegister()) {
670 			RegisterArg arg = (RegisterArg) retArg;
671 			int regNum = arg.getRegNum();
672 			for (BlockNode pred : preds) {
673 				for (InsnNode insnNode : pred.getInstructions()) {
674 					RegisterArg result = insnNode.getResult();
675 					if (result != null && result.getRegNum() == regNum) {
676 						return true;
677 					}
678 				}
679 			}
680 		}
681 		return false;
682 	}

701 	private static void removeUnreachableBlocks(MethodNode mth) {
702 		Set<BlockNode> toRemove = null;
703 		for (BlockNode block : mth.getBasicBlocks()) {
704 			if (block.getPredecessors().isEmpty() && block != mth.getEnterBlock()) {
705 				toRemove = new LinkedHashSet<>();
706 				BlockSplitter.collectSuccessors(block, mth.getEnterBlock(), toRemove);
707 			}
708 		}
709 		if (toRemove == null || toRemove.isEmpty()) {
710 			return;
711 		}
712 
713 		toRemove.forEach(BlockSplitter::detachBlock);
714 		mth.getBasicBlocks().removeAll(toRemove);
715 		long notEmptyBlocks = toRemove.stream().filter(block -> !block.getInstructions().isEmpty()).count();
716 		if (notEmptyBlocks != 0) {
717 			int insnsCount = toRemove.stream().mapToInt(block -> block.getInstructions().size()).sum();
718 			mth.addWarnComment("Unreachable blocks removed: " + notEmptyBlocks + ", instructions: " + insnsCount);
719 		}
720 	}
