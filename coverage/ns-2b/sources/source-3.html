43 	public void visit(MethodNode mth) {
44 		if (mth.isNoCode() || mth.isNoExceptionHandlers()) {
45 			return;
46 		}
47 		try {
48 			boolean applied = false;
49 			List<TryCatchBlockAttr> tryBlocks = mth.getAll(AType.TRY_BLOCKS_LIST);
50 			for (TryCatchBlockAttr tryBlock : tryBlocks) {
51 				applied |= processTryBlock(mth, tryBlock);
52 			}
53 			if (applied) {

56 				List<TryCatchBlockAttr> clearedTryBlocks = new ArrayList<>(tryBlocks);
57 				if (clearedTryBlocks.removeIf(tb -> tb.isMerged() || tb.getHandlers().isEmpty())) {
58 					mth.remove(AType.TRY_BLOCKS_LIST);
59 					mth.addAttr(AType.TRY_BLOCKS_LIST, clearedTryBlocks);
60 				}
61 			}
62 		} catch (Exception e) {

64 			undoFinallyVisitor(mth);
65 		}

69 		if (tryBlock.isMerged()) {
70 			return false;
71 		}
72 		ExceptionHandler allHandler = null;
73 		InsnNode reThrowInsn = null;
74 		for (ExceptionHandler excHandler : tryBlock.getHandlers()) {
75 			if (excHandler.isCatchAll()) {
76 				allHandler = excHandler;
77 				for (BlockNode excBlock : excHandler.getBlocks()) {
78 					InsnNode lastInsn = BlockUtils.getLastInsn(excBlock);
79 					if (lastInsn != null && lastInsn.getType() == InsnType.THROW) {
80 						reThrowInsn = BlockUtils.getLastInsn(excBlock);
81 					}
82 				}
83 				break;
84 			}
85 		}
86 		if (allHandler != null && reThrowInsn != null) {
87 			if (extractFinally(mth, tryBlock, allHandler)) {

89 				return true;
90 			}
91 		}
92 		return false;

98 	private static boolean extractFinally(MethodNode mth, TryCatchBlockAttr tryBlock, ExceptionHandler allHandler) {
99 		BlockNode handlerBlock = allHandler.getHandlerBlock();
100 		List<BlockNode> handlerBlocks =
101 				new ArrayList<>(BlockUtils.collectBlocksDominatedByWithExcHandlers(mth, handlerBlock, handlerBlock));
102 		handlerBlocks.remove(handlerBlock); // exclude block with 'move-exception'
103 		handlerBlocks.removeIf(b -> BlockUtils.checkLastInsnType(b, InsnType.THROW));
104 		if (handlerBlocks.isEmpty() || BlockUtils.isAllBlocksEmpty(handlerBlocks)) {
105 			// remove empty catch
106 			allHandler.getTryBlock().removeHandler(allHandler);
107 			return true;
108 		}
109 		BlockNode startBlock = Utils.getOne(handlerBlock.getCleanSuccessors());
110 		FinallyExtractInfo extractInfo = new FinallyExtractInfo(allHandler, startBlock, handlerBlocks);
111 
112 		boolean hasInnerBlocks = !tryBlock.getInnerTryBlocks().isEmpty();
113 		List<ExceptionHandler> handlers;
114 		if (hasInnerBlocks) {
115 			// collect handlers from this and all inner blocks (intentionally not using recursive collect for
116 			// now)
117 			handlers = new ArrayList<>(tryBlock.getHandlers());
118 			for (TryCatchBlockAttr innerTryBlock : tryBlock.getInnerTryBlocks()) {
119 				handlers.addAll(innerTryBlock.getHandlers());
120 			}
121 		} else {
122 			handlers = tryBlock.getHandlers();
123 		}
124 		if (handlers.isEmpty()) {
125 			return false;
126 		}
127 		// search 'finally' instructions in other handlers
128 		for (ExceptionHandler otherHandler : handlers) {
129 			if (otherHandler == allHandler) {
130 				continue;
131 			}
132 			for (BlockNode checkBlock : otherHandler.getBlocks()) {
133 				if (searchDuplicateInsns(checkBlock, extractInfo)) {
134 					break;
135 				} else {
136 					extractInfo.getFinallyInsnsSlice().resetIncomplete();
137 				}
138 			}
139 		}
140 		boolean mergeInnerTryBlocks;
141 		int duplicatesCount = extractInfo.getDuplicateSlices().size();
142 		boolean fullTryBlock = duplicatesCount == (handlers.size() - 1);
143 		if (fullTryBlock) {
144 			// all collected handlers have duplicate block
145 			mergeInnerTryBlocks = hasInnerBlocks;
146 		} else {
147 			// some handlers don't have duplicated blocks
148 			if (!hasInnerBlocks || duplicatesCount != (tryBlock.getHandlers().size() - 1)) {
149 				// unexpected count of duplicated slices
150 				return false;
151 			}
152 			mergeInnerTryBlocks = false;
153 		}
154 
155 		// remove 'finally' from 'try' blocks, check all up paths on each exit (connected with finally exit)
156 		List<BlockNode> tryBlocks = allHandler.getTryBlock().getBlocks();
157 		BlockNode bottomFinallyBlock = BlockUtils.followEmptyPath(BlockUtils.getBottomBlock(allHandler.getBlocks()));
158 		BlockNode bottom = BlockUtils.getNextBlock(bottomFinallyBlock);
159 		if (bottom == null) {
160 			return false;
161 		}
162 		boolean found = false;
163 		List<BlockNode> pathBlocks = getPathStarts(mth, bottom, bottomFinallyBlock);
164 		for (BlockNode pred : pathBlocks) {
165 			List<BlockNode> upPath = BlockUtils.collectPredecessors(mth, pred, tryBlocks);
166 			if (upPath.size() < handlerBlocks.size()) {
167 				continue;
168 			}
169 			for (BlockNode block : upPath) {
170 				if (searchDuplicateInsns(block, extractInfo)) {
171 					found = true;
172 					break;
173 				} else {
174 					extractInfo.getFinallyInsnsSlice().resetIncomplete();
175 				}
176 			}
177 		}
178 		if (!found) {
179 			return false;
180 		}
181 		if (!checkSlices(extractInfo)) {
182 			mth.addWarnComment("Finally extract failed");
183 			return false;
184 		}
185 
186 		// 'finally' extract confirmed, apply
187 		apply(extractInfo);
188 		allHandler.setFinally(true);
189 
190 		if (mergeInnerTryBlocks) {
191 			List<TryCatchBlockAttr> innerTryBlocks = tryBlock.getInnerTryBlocks();
192 			for (TryCatchBlockAttr innerTryBlock : innerTryBlocks) {
193 				tryBlock.getHandlers().addAll(innerTryBlock.getHandlers());
194 				tryBlock.getBlocks().addAll(innerTryBlock.getBlocks());
195 				innerTryBlock.setMerged(true);
196 			}
197 			tryBlock.setBlocks(ListUtils.distinctList(tryBlock.getBlocks()));
198 			innerTryBlocks.clear();
199 		}
200 		return true;
201 	}

212 		InsnsSlice finallySlice = extractInfo.getFinallyInsnsSlice();
213 		List<InsnNode> finallyInsnsList = finallySlice.getInsnsList();
214 
215 		for (InsnsSlice dupSlice : extractInfo.getDuplicateSlices()) {
216 			List<InsnNode> dupInsnsList = dupSlice.getInsnsList();
217 			if (dupInsnsList.size() != finallyInsnsList.size()) {

220 				}

222 			}
223 		}

225 			InsnNode finallyInsn = finallyInsnsList.get(i);

232 					}
233 					return false;
234 				}
235 			}
236 		}

241 		markSlice(extractInfo.getFinallyInsnsSlice(), AFlag.FINALLY_INSNS);
242 		for (InsnsSlice dupSlice : extractInfo.getDuplicateSlices()) {
243 			markSlice(dupSlice, AFlag.DONT_GENERATE);
244 		}
245 		InsnsSlice finallySlice = extractInfo.getFinallyInsnsSlice();
246 		List<InsnNode> finallyInsnsList = finallySlice.getInsnsList();
247 		for (int i = 0; i < finallyInsnsList.size(); i++) {
248 			InsnNode finallyInsn = finallyInsnsList.get(i);
249 			for (InsnsSlice dupSlice : extractInfo.getDuplicateSlices()) {
250 				InsnNode dupInsn = dupSlice.getInsnsList().get(i);
251 				copyCodeVars(finallyInsn, dupInsn);
252 			}
253 		}

338 	private static InsnsSlice isStartBlock(BlockNode dupBlock, BlockNode finallyBlock, FinallyExtractInfo extractInfo) {
339 		List<InsnNode> dupInsns = dupBlock.getInstructions();
340 		List<InsnNode> finallyInsns = finallyBlock.getInstructions();
341 		if (dupInsns.size() < finallyInsns.size()) {
342 			return null;
343 		}
344 		int startPos = dupInsns.size() - finallyInsns.size();
345 		int endPos = 0;
346 		// fast check from end of block
347 		if (!checkInsns(dupInsns, finallyInsns, startPos)) {
348 			// check from block start
349 			if (checkInsns(dupInsns, finallyInsns, 0)) {
350 				startPos = 0;
351 				endPos = finallyInsns.size();
352 			} else {
353 				// search start insn
354 				boolean found = false;
355 				for (int i = 1; i < startPos; i++) {
356 					if (checkInsns(dupInsns, finallyInsns, i)) {
357 						startPos = i;
358 						endPos = finallyInsns.size() + i;
359 						found = true;
360 						break;
361 					}
362 				}
363 				if (!found) {
364 					return null;
365 				}
366 			}
367 		}
368 
369 		// put instructions into slices
370 		boolean complete;
371 		InsnsSlice slice = new InsnsSlice();
372 		int endIndex;
373 		if (endPos != 0) {
374 			endIndex = endPos + 1;
375 			// both slices completed
376 			complete = true;
377 		} else {
378 			endIndex = dupInsns.size();
379 			complete = false;
380 		}
381 
382 		// fill dup insns slice
383 		for (int i = startPos; i < endIndex; i++) {
384 			slice.addInsn(dupInsns.get(i), dupBlock);
385 		}
386 
387 		// fill finally insns slice
388 		InsnsSlice finallySlice = extractInfo.getFinallyInsnsSlice();
389 		if (finallySlice.isComplete()) {
390 			// compare slices
391 			if (finallySlice.getInsnsList().size() != slice.getInsnsList().size()) {
392 				if (LOG.isDebugEnabled()) {
393 					LOG.debug("Another duplicated slice has different insns count: {}, finally: {}", slice, finallySlice);
394 				}
395 				return null;
396 			}
397 			// TODO: add additional slices checks
398 			// and try to extract common part if found difference
399 		} else {
400 			for (InsnNode finallyInsn : finallyInsns) {
401 				finallySlice.addInsn(finallyInsn, finallyBlock);
402 			}
403 		}
404 
405 		if (complete) {
406 			slice.setComplete(true);
407 			finallySlice.setComplete(true);
408 		}
409 		return slice;
410 	}

413 		for (int i = finallyInsns.size() - 1; i >= 0; i--) {

423 	private static boolean checkBlocksTree(BlockNode dupBlock, BlockNode finallyBlock,
424 			InsnsSlice dupSlice, FinallyExtractInfo extractInfo) {
425 		InsnsSlice finallySlice = extractInfo.getFinallyInsnsSlice();
426 
427 		List<BlockNode> finallyCS = getSuccessorsWithoutLoop(finallyBlock);
428 		List<BlockNode> dupCS = getSuccessorsWithoutLoop(dupBlock);
429 		if (finallyCS.size() == dupCS.size()) {
430 			for (int i = 0; i < finallyCS.size(); i++) {
431 				BlockNode finSBlock = finallyCS.get(i);
432 				BlockNode dupSBlock = dupCS.get(i);
433 				if (extractInfo.getAllHandlerBlocks().contains(finSBlock)) {
434 					if (!compareBlocks(dupSBlock, finSBlock, dupSlice, extractInfo)) {
435 						return false;
436 					}
437 					if (!checkBlocksTree(dupSBlock, finSBlock, dupSlice, extractInfo)) {
438 						return false;
439 					}
440 					dupSlice.addBlock(dupSBlock);
441 					finallySlice.addBlock(finSBlock);
442 				}
443 			}
444 		}
445 		dupSlice.setComplete(true);
446 		finallySlice.setComplete(true);
447 		return true;
448 	}

457 	private static boolean compareBlocks(BlockNode dupBlock, BlockNode finallyBlock, InsnsSlice dupSlice, FinallyExtractInfo extractInfo) {
458 		List<InsnNode> dupInsns = dupBlock.getInstructions();
459 		List<InsnNode> finallyInsns = finallyBlock.getInstructions();
460 		int dupInsnCount = dupInsns.size();
461 		int finallyInsnCount = finallyInsns.size();
462 		if (finallyInsnCount == 0) {
463 			return dupInsnCount == 0;
464 		}
465 		if (dupInsnCount < finallyInsnCount) {
466 			return false;
467 		}
468 		for (int i = 0; i < finallyInsnCount; i++) {
469 			if (!sameInsns(dupInsns.get(i), finallyInsns.get(i))) {
470 				return false;
471 			}
472 		}
473 		if (dupInsnCount > finallyInsnCount) {
474 			dupSlice.addInsns(dupBlock, 0, finallyInsnCount);
475 			dupSlice.setComplete(true);
476 			InsnsSlice finallyInsnsSlice = extractInfo.getFinallyInsnsSlice();
477 			finallyInsnsSlice.addBlock(finallyBlock);
478 			finallyInsnsSlice.setComplete(true);
479 		}
480 		return true;
481 	}

483 	private static boolean sameInsns(InsnNode remInsn, InsnNode fInsn) {
484 		if (!remInsn.isSame(fInsn)) {
485 			return false;
486 		}
487 		// TODO: check instance arg in ConstructorInsn
488 		// TODO: compare literals
489 		for (int i = 0; i < remInsn.getArgsCount(); i++) {
490 			InsnArg remArg = remInsn.getArg(i);
491 			InsnArg fArg = fInsn.getArg(i);
492 			if (remArg.isRegister() != fArg.isRegister()) {
493 				return false;
494 			}
495 			boolean remConst = remArg.isConst();
496 			if (remConst != fArg.isConst()) {
497 				return false;
498 			}
499 			if (remConst && !remArg.isSameConst(fArg)) {
500 				return false;
501 			}
502 		}
503 		return true;
504 	}
