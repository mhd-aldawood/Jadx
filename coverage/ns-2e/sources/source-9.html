287 	private LoopRegion makeLoopRegion(IRegion curRegion, LoopInfo loop, List<BlockNode> exitBlocks) {
288 		for (BlockNode block : exitBlocks) {
289 			if (block.contains(AType.EXC_HANDLER)) {
290 				continue;
291 			}
292 			InsnNode lastInsn = BlockUtils.getLastInsn(block);
293 			if (lastInsn == null || lastInsn.getType() != InsnType.IF) {
294 				continue;
295 			}
296 			List<LoopInfo> loops = block.getAll(AType.LOOP);
297 			if (!loops.isEmpty() && loops.get(0) != loop) {
298 				// skip nested loop condition
299 				continue;
300 			}
301 			LoopRegion loopRegion = new LoopRegion(curRegion, loop, block, block == loop.getEnd());
302 			boolean found;
303 			if (block == loop.getStart() || block == loop.getEnd()
304 					|| BlockUtils.isEmptySimplePath(loop.getStart(), block)) {
305 				found = true;
306 			} else if (block.getPredecessors().contains(loop.getStart())) {
307 				loopRegion.setPreCondition(loop.getStart());
308 				// if we can't merge pre-condition this is not correct header
309 				found = loopRegion.checkPreCondition();
310 			} else {
311 				found = false;
312 			}
313 			if (found) {
314 				List<LoopInfo> list = mth.getAllLoopsForBlock(block);
315 				if (list.size() >= 2) {
316 					// bad condition if successors going out of all loops
317 					boolean allOuter = true;
318 					for (BlockNode outerBlock : block.getCleanSuccessors()) {
319 						List<LoopInfo> outLoopList = mth.getAllLoopsForBlock(outerBlock);
320 						outLoopList.remove(loop);
321 						if (!outLoopList.isEmpty()) {
322 							// goes to outer loop
323 							allOuter = false;
324 							break;
325 						}
326 					}
327 					if (allOuter) {
328 						found = false;
329 					}
330 				}
331 			}
332 			if (found && !checkLoopExits(loop, block)) {
333 				found = false;
334 			}
335 			if (found) {
336 				return loopRegion;
337 			}
338 		}
339 		// no exit found => endless loop
340 		return null;
341 	}

461 		boolean confirm = false;

463 		// 1. jump to outer loop
464 		BlockNode exitEnd = BlockUtils.followEmptyPath(exit);
465 		List<LoopInfo> loops = exitEnd.getAll(AType.LOOP);
466 		for (LoopInfo loopAtEnd : loops) {
467 			if (loopAtEnd != loop) {

471 			}
472 		}

550 			Set<BlockNode> loopExitNodes) {
551 		if (!pred.contains(AFlag.SYNTHETIC)
552 				|| BlockUtils.checkLastInsnType(pred, InsnType.CONTINUE)) {
553 			return false;
554 		}
555 		List<BlockNode> preds = pred.getPredecessors();
556 		if (preds.isEmpty()) {
557 			return false;
558 		}
559 		BlockNode codePred = preds.get(0);
560 		if (codePred.contains(AFlag.ADDED_TO_REGION)) {
561 			return false;
562 		}
563 		if (loopEnd.isDominator(codePred)
564 				|| loopExitNodes.contains(codePred)) {
565 			return false;
566 		}
567 		if (isDominatedOnBlocks(codePred, predecessors)) {
568 			return false;
569 		}
570 		boolean gotoExit = false;
571 		for (BlockNode exit : loopExitNodes) {
572 			if (BlockUtils.isPathExists(codePred, exit)) {
573 				gotoExit = true;
574 				break;

589 	private BlockNode processMonitorEnter(IRegion curRegion, BlockNode block, InsnNode insn, RegionStack stack) {
590 		SynchronizedRegion synchRegion = new SynchronizedRegion(curRegion, insn);
591 		synchRegion.getSubBlocks().add(block);
592 		curRegion.getSubBlocks().add(synchRegion);
593 
594 		Set<BlockNode> exits = new LinkedHashSet<>();
595 		Set<BlockNode> cacheSet = new HashSet<>();
596 		traverseMonitorExits(synchRegion, insn.getArg(0), block, exits, cacheSet);
597 
598 		for (InsnNode exitInsn : synchRegion.getExitInsns()) {
599 			BlockNode insnBlock = BlockUtils.getBlockByInsn(mth, exitInsn);
600 			if (insnBlock != null) {
601 				insnBlock.add(AFlag.DONT_GENERATE);
602 			}
603 			// remove arg from MONITOR_EXIT to allow inline in MONITOR_ENTER
604 			exitInsn.removeArg(0);
605 			exitInsn.add(AFlag.DONT_GENERATE);
606 		}
607 
608 		BlockNode body = getNextBlock(block);
609 		if (body == null) {
610 			mth.addWarn("Unexpected end of synchronized block");
611 			return null;
612 		}
613 		BlockNode exit = null;
614 		if (exits.size() == 1) {
615 			exit = getNextBlock(exits.iterator().next());
616 		} else if (exits.size() > 1) {
617 			cacheSet.clear();
618 			exit = traverseMonitorExitsCross(body, exits, cacheSet);
619 		}
620 
621 		stack.push(synchRegion);
622 		if (exit != null) {
623 			stack.addExit(exit);
624 		} else {
625 			for (BlockNode exitBlock : exits) {
626 				// don't add exit blocks which leads to method end blocks ('return', 'throw', etc)
627 				List<BlockNode> list = BlockUtils.buildSimplePath(exitBlock);
628 				if (list.isEmpty() || !BlockUtils.isExitBlock(mth, Utils.last(list))) {
629 					stack.addExit(exitBlock);
630 					// we can still try using this as an exit block to make sure it's visited.
631 					exit = exitBlock;
632 				}
633 			}
634 		}
635 		synchRegion.getSubBlocks().add(makeRegion(body, stack));
636 		stack.pop();
637 		return exit;
638 	}

641 	 * Traverse from monitor-enter thru successors and collect blocks contains monitor-exit
642 	 */
643 	private static void traverseMonitorExits(SynchronizedRegion region, InsnArg arg, BlockNode block, Set<BlockNode> exits,
644 			Set<BlockNode> visited) {
645 		visited.add(block);
646 		for (InsnNode insn : block.getInstructions()) {
647 			if (insn.getType() == InsnType.MONITOR_EXIT
648 					&& insn.getArgsCount() > 0
649 					&& insn.getArg(0).equals(arg)) {
650 				exits.add(block);
651 				region.getExitInsns().add(insn);
652 				return;
653 			}
654 		}
655 		for (BlockNode node : block.getSuccessors()) {
656 			if (!visited.contains(node)) {
657 				traverseMonitorExits(region, arg, node, exits, visited);
658 			}
659 		}
660 	}

663 	 * Traverse from monitor-enter thru successors and search for exit paths cross
664 	 */
665 	private static BlockNode traverseMonitorExitsCross(BlockNode block, Set<BlockNode> exits, Set<BlockNode> visited) {
666 		visited.add(block);
667 		for (BlockNode node : block.getCleanSuccessors()) {
668 			boolean cross = true;
669 			for (BlockNode exitBlock : exits) {
670 				boolean p = isPathExists(exitBlock, node);
671 				if (!p) {
672 					cross = false;
673 					break;
674 				}
675 			}
676 			if (cross) {
677 				return node;
678 			}
679 			if (!visited.contains(node)) {
680 				BlockNode res = traverseMonitorExitsCross(node, exits, visited);
681 				if (res != null) {
682 					return res;
683 				}
684 			}
685 		}
686 		return null;
687 	}

689 	private BlockNode processIf(IRegion currentRegion, BlockNode block, IfNode ifnode, RegionStack stack) {
690 		if (block.contains(AFlag.ADDED_TO_REGION)) {
691 			// block already included in other 'if' region
692 			return ifnode.getThenBlock();
693 		}
694 
695 		IfInfo currentIf = makeIfInfo(mth, block);
696 		if (currentIf == null) {
697 			return null;
698 		}
699 		IfInfo mergedIf = mergeNestedIfNodes(currentIf);
700 		if (mergedIf != null) {
701 			currentIf = mergedIf;
702 		} else {
703 			// invert simple condition (compiler often do it)
704 			currentIf = IfInfo.invert(currentIf);
705 		}
706 		IfInfo modifiedIf = IfMakerHelper.restructureIf(mth, block, currentIf);
707 		if (modifiedIf != null) {
708 			currentIf = modifiedIf;
709 		} else {
710 			if (currentIf.getMergedBlocks().size() <= 1) {
711 				return null;
712 			}
713 			currentIf = makeIfInfo(mth, block);
714 			currentIf = IfMakerHelper.restructureIf(mth, block, currentIf);
715 			if (currentIf == null) {
716 				// all attempts failed
717 				return null;
718 			}
719 		}
720 		confirmMerge(currentIf);
721 
722 		IfRegion ifRegion = new IfRegion(currentRegion);
723 		ifRegion.updateCondition(currentIf);
724 		currentRegion.getSubBlocks().add(ifRegion);
725 
726 		BlockNode outBlock = currentIf.getOutBlock();
727 		stack.push(ifRegion);
728 		stack.addExit(outBlock);
729 
730 		ifRegion.setThenRegion(makeRegion(currentIf.getThenBlock(), stack));
731 		BlockNode elseBlock = currentIf.getElseBlock();
732 		if (elseBlock == null || stack.containsExit(elseBlock)) {
733 			ifRegion.setElseRegion(null);
734 		} else {
735 			ifRegion.setElseRegion(makeRegion(elseBlock, stack));
736 		}
737 
738 		// insert edge insns in new 'else' branch
739 		// TODO: make more common algorithm
740 		if (ifRegion.getElseRegion() == null && outBlock != null) {
741 			List<EdgeInsnAttr> edgeInsnAttrs = outBlock.getAll(AType.EDGE_INSN);
742 			if (!edgeInsnAttrs.isEmpty()) {
743 				Region elseRegion = new Region(ifRegion);
744 				for (EdgeInsnAttr edgeInsnAttr : edgeInsnAttrs) {
745 					if (edgeInsnAttr.getEnd().equals(outBlock)) {
746 						addEdgeInsn(currentIf, elseRegion, edgeInsnAttr);
747 					}
748 				}
749 				ifRegion.setElseRegion(elseRegion);
750 			}
751 		}
752 
753 		stack.pop();
754 		return outBlock;
755 	}

757 	private void addEdgeInsn(IfInfo ifInfo, Region region, EdgeInsnAttr edgeInsnAttr) {
758 		BlockNode start = edgeInsnAttr.getStart();
759 		boolean fromThisIf = false;

762 				fromThisIf = true;

764 			}
765 		}
766 		if (!fromThisIf) {
767 			return;
768 		}

772 	private BlockNode processSwitch(IRegion currentRegion, BlockNode block, SwitchInsn insn, RegionStack stack) {
773 		// map case blocks to keys
774 		int len = insn.getTargets().length;
775 		Map<BlockNode, List<Object>> blocksMap = new LinkedHashMap<>(len);
776 		BlockNode[] targetBlocksArr = insn.getTargetBlocks();
777 		for (int i = 0; i < len; i++) {
778 			List<Object> keys = blocksMap.computeIfAbsent(targetBlocksArr[i], k -> new ArrayList<>(2));
779 			keys.add(insn.getKey(i));
780 		}
781 		BlockNode defCase = insn.getDefTargetBlock();
782 		if (defCase != null) {
783 			List<Object> keys = blocksMap.computeIfAbsent(defCase, k -> new ArrayList<>(1));
784 			keys.add(SwitchRegion.DEFAULT_CASE_KEY);
785 		}
786 
787 		// search 'out' block - 'next' block after whole switch statement
788 		BlockNode out;
789 		LoopInfo loop = mth.getLoopForBlock(block);
790 		if (loop == null) {
791 			out = calcPostDomOut(mth, block, mth.getPreExitBlocks());
792 		} else {
793 			BlockNode loopEnd = loop.getEnd();
794 			stack.addExit(loop.getStart());
795 			if (stack.containsExit(block)
796 					|| block == loopEnd
797 					|| loopEnd.getPredecessors().contains(block)) {
798 				// in exits or last insn in loop => no 'out' block
799 				out = null;
800 			} else {
801 				// treat 'continue' as exit
802 				out = calcPostDomOut(mth, block, loopEnd.getPredecessors());
803 				if (out != null) {
804 					insertContinueInSwitch(block, out, loopEnd);
805 				} else {
806 					// no 'continue'
807 					out = calcPostDomOut(mth, block, Collections.singletonList(loopEnd));
808 				}
809 			}
810 			if (out == loop.getStart()) {
811 				// no other outs instead back edge to loop start
812 				out = null;
813 			}
814 		}
815 		if (out != null && processedBlocks.get(out.getId())) {
816 			// out block already processed, prevent endless loop
817 			throw new JadxRuntimeException("Failed to find switch 'out' block");
818 		}
819 
820 		SwitchRegion sw = new SwitchRegion(currentRegion, block);
821 		currentRegion.getSubBlocks().add(sw);
822 		stack.push(sw);
823 		stack.addExit(out);
824 
825 		// detect fallthrough cases
826 		Map<BlockNode, BlockNode> fallThroughCases = new LinkedHashMap<>();
827 		if (out != null) {
828 			BitSet caseBlocks = BlockUtils.blocksToBitSet(mth, blocksMap.keySet());
829 			caseBlocks.clear(out.getId());
830 			for (BlockNode successor : block.getCleanSuccessors()) {
831 				BlockNode fallThroughBlock = searchFallThroughCase(successor, out, caseBlocks);
832 				if (fallThroughBlock != null) {
833 					fallThroughCases.put(successor, fallThroughBlock);
834 				}
835 			}
836 			// check fallthrough cases order
837 			if (!fallThroughCases.isEmpty() && isBadCasesOrder(blocksMap, fallThroughCases)) {
838 				Map<BlockNode, List<Object>> newBlocksMap = reOrderSwitchCases(blocksMap, fallThroughCases);
839 				if (isBadCasesOrder(newBlocksMap, fallThroughCases)) {
840 					mth.addWarnComment("Can't fix incorrect switch cases order, some code will duplicate");
841 					fallThroughCases.clear();
842 				} else {
843 					blocksMap = newBlocksMap;
844 				}
845 			}
846 		}
847 
848 		for (Entry<BlockNode, List<Object>> entry : blocksMap.entrySet()) {
849 			List<Object> keysList = entry.getValue();
850 			BlockNode caseBlock = entry.getKey();
851 			if (stack.containsExit(caseBlock)) {
852 				sw.addCase(keysList, new Region(stack.peekRegion()));
853 			} else {
854 				BlockNode next = fallThroughCases.get(caseBlock);
855 				stack.addExit(next);
856 				Region caseRegion = makeRegion(caseBlock, stack);
857 				stack.removeExit(next);
858 				if (next != null) {
859 					next.add(AFlag.FALL_THROUGH);
860 					caseRegion.add(AFlag.FALL_THROUGH);
861 				}
862 				sw.addCase(keysList, caseRegion);
863 				// 'break' instruction will be inserted in RegionMakerVisitor.PostRegionVisitor
864 			}
865 		}
866 
867 		removeEmptyCases(insn, sw, defCase);
868 
869 		stack.pop();
870 		return out;
871 	}

873 	@Nullable
874 	private BlockNode searchFallThroughCase(BlockNode successor, BlockNode out, BitSet caseBlocks) {
875 		BitSet df = successor.getDomFrontier();
876 		if (df.intersects(caseBlocks)) {
877 			return getOneIntersectionBlock(out, caseBlocks, df);
878 		}
879 		Set<BlockNode> allPathsBlocks = BlockUtils.getAllPathsBlocks(successor, out);
880 		Map<BlockNode, BitSet> bitSetMap = BlockUtils.calcPartialPostDominance(mth, allPathsBlocks, out);
881 		BitSet pdoms = bitSetMap.get(successor);
882 		if (pdoms != null && pdoms.intersects(caseBlocks)) {
883 			return getOneIntersectionBlock(out, caseBlocks, pdoms);
884 		}
885 		return null;
886 	}

896 	@Nullable
897 	private static BlockNode calcPostDomOut(MethodNode mth, BlockNode block, List<BlockNode> exits) {
898 		if (exits.size() == 1 && mth.getExitBlock().equals(exits.get(0))) {
899 			// simple case: for only one exit which is equal to method exit block
900 			return BlockUtils.calcImmediatePostDominator(mth, block);
901 		}
902 		// fast search: union of blocks dominance frontier
903 		// work if no fallthrough cases and no returns inside switch
904 		BitSet outs = BlockUtils.copyBlocksBitSet(mth, block.getDomFrontier());
905 		for (BlockNode s : block.getCleanSuccessors()) {
906 			outs.or(s.getDomFrontier());
907 		}
908 		outs.clear(block.getId());
909 
910 		if (outs.cardinality() != 1) {
911 			// slow search: calculate partial post-dominance for every exit node
912 			BitSet ipdoms = BlockUtils.newBlocksBitSet(mth);
913 			for (BlockNode exitBlock : exits) {
914 				if (BlockUtils.isAnyPathExists(block, exitBlock)) {
915 					Set<BlockNode> pathBlocks = BlockUtils.getAllPathsBlocks(block, exitBlock);
916 					BlockNode ipdom = BlockUtils.calcPartialImmediatePostDominator(mth, block, pathBlocks, exitBlock);
917 					if (ipdom != null) {
918 						ipdoms.set(ipdom.getId());
919 					}
920 				}
921 			}
922 			outs.and(ipdoms);
923 		}
924 		return BlockUtils.bitSetToOneBlock(mth, outs);
925 	}

928 	 * Remove empty case blocks:
929 	 * 1. single 'default' case
930 	 * 2. filler cases if switch is 'packed' and 'default' case is empty
931 	 */
932 	private void removeEmptyCases(SwitchInsn insn, SwitchRegion sw, BlockNode defCase) {
933 		boolean defaultCaseIsEmpty;
934 		if (defCase == null) {
935 			defaultCaseIsEmpty = true;
936 		} else {
937 			defaultCaseIsEmpty = sw.getCases().stream()
938 					.anyMatch(c -> c.getKeys().contains(SwitchRegion.DEFAULT_CASE_KEY)
939 							&& RegionUtils.isEmpty(c.getContainer()));
940 		}
941 		if (defaultCaseIsEmpty) {
942 			sw.getCases().removeIf(caseInfo -> {
943 				if (RegionUtils.isEmpty(caseInfo.getContainer())) {
944 					List<Object> keys = caseInfo.getKeys();
945 					if (keys.contains(SwitchRegion.DEFAULT_CASE_KEY)) {
946 						return true;
947 					}
948 					if (insn.isPacked()) {
949 						return true;
950 					}
951 				}
952 				return false;
953 			});
954 		}
955 	}

957 	private boolean isBadCasesOrder(Map<BlockNode, List<Object>> blocksMap, Map<BlockNode, BlockNode> fallThroughCases) {
958 		BlockNode nextCaseBlock = null;
959 		for (BlockNode caseBlock : blocksMap.keySet()) {
960 			if (nextCaseBlock != null && !caseBlock.equals(nextCaseBlock)) {
961 				return true;
962 			}
963 			nextCaseBlock = fallThroughCases.get(caseBlock);
964 		}
965 		return nextCaseBlock != null;

968 	private Map<BlockNode, List<Object>> reOrderSwitchCases(Map<BlockNode, List<Object>> blocksMap,
969 			Map<BlockNode, BlockNode> fallThroughCases) {
970 		List<BlockNode> list = new ArrayList<>(blocksMap.size());
971 		list.addAll(blocksMap.keySet());
972 		list.sort((a, b) -> {
973 			BlockNode nextA = fallThroughCases.get(a);
974 			if (nextA != null) {
975 				if (b.equals(nextA)) {
976 					return -1;
977 				}
978 			} else if (a.equals(fallThroughCases.get(b))) {
979 				return 1;
980 			}
981 			return 0;
982 		});
983 
984 		Map<BlockNode, List<Object>> newBlocksMap = new LinkedHashMap<>(blocksMap.size());
985 		for (BlockNode key : list) {
986 			newBlocksMap.put(key, blocksMap.get(key));
987 		}
988 		return newBlocksMap;
989 	}

1009 	public IRegion processTryCatchBlocks(MethodNode mth) {
1010 		List<TryCatchBlockAttr> tcs = mth.getAll(AType.TRY_BLOCKS_LIST);
1011 		for (TryCatchBlockAttr tc : tcs) {
1012 			List<BlockNode> blocks = new ArrayList<>(tc.getHandlersCount());
1013 			Set<BlockNode> splitters = new HashSet<>();
1014 			for (ExceptionHandler handler : tc.getHandlers()) {
1015 				BlockNode handlerBlock = handler.getHandlerBlock();
1016 				if (handlerBlock != null) {
1017 					blocks.add(handlerBlock);
1018 					splitters.add(BlockUtils.getTopSplitterForHandler(handlerBlock));
1019 				} else {
1020 					mth.addDebugComment("No exception handler block: " + handler);
1021 				}
1022 			}
1023 			Set<BlockNode> exits = new HashSet<>();
1024 			for (BlockNode splitter : splitters) {
1025 				for (BlockNode handler : blocks) {
1026 					if (handler.contains(AFlag.REMOVE)) {
1027 						continue;
1028 					}
1029 					List<BlockNode> s = splitter.getSuccessors();
1030 					if (s.isEmpty()) {
1031 						mth.addDebugComment("No successors for splitter: " + splitter);
1032 						continue;
1033 					}
1034 					BlockNode ss = s.get(0);
1035 					BlockNode cross = BlockUtils.getPathCross(mth, ss, handler);
1036 					if (cross != null && cross != ss && cross != handler) {
1037 						exits.add(cross);
1038 					}
1039 				}
1040 			}
1041 			for (ExceptionHandler handler : tc.getHandlers()) {
1042 				processExcHandler(mth, handler, exits);
1043 			}
1044 		}
1045 		return processHandlersOutBlocks(mth, tcs);
1046 	}
