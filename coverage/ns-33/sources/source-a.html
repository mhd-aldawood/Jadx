197 	private boolean calculateFromBounds(MethodNode mth, SSAVar ssaVar) {
198 		TypeInfo typeInfo = ssaVar.getTypeInfo();
199 		Set<ITypeBound> bounds = typeInfo.getBounds();
200 		Optional<ArgType> bestTypeOpt = selectBestTypeFromBounds(bounds);
201 		if (!bestTypeOpt.isPresent()) {
202 			if (Consts.DEBUG_TYPE_INFERENCE) {
203 				LOG.warn("Failed to select best type from bounds, count={} : ", bounds.size());
204 				for (ITypeBound bound : bounds) {
205 					LOG.warn("  {}", bound);
206 				}
207 			}
208 			return false;
209 		}
210 		ArgType candidateType = bestTypeOpt.get();
211 		TypeUpdateResult result = typeUpdate.apply(mth, ssaVar, candidateType);
212 		if (result == TypeUpdateResult.REJECT) {
213 			if (Consts.DEBUG_TYPE_INFERENCE) {
214 				if (ssaVar.getTypeInfo().getType().equals(candidateType)) {
215 					LOG.info("Same type rejected: {} -> {}, bounds: {}", ssaVar, candidateType, bounds);
216 				} else if (candidateType.isTypeKnown()) {
217 					LOG.debug("Type rejected: {} -> {}, bounds: {}", ssaVar, candidateType, bounds);
218 				}
219 			}
220 			return false;
221 		}
222 		return result == TypeUpdateResult.CHANGED;
223 	}

263 	private void addAssignBound(TypeInfo typeInfo, RegisterArg assign) {
264 		ArgType immutableType = assign.getImmutableType();
265 		if (immutableType != null) {
266 			addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, immutableType));
267 			return;
268 		}
269 		InsnNode insn = assign.getParentInsn();
270 		if (insn == null || insn.getResult() == null) {
271 			addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, assign.getInitType()));
272 			return;
273 		}
274 		switch (insn.getType()) {
275 			case NEW_INSTANCE:
276 				ArgType clsType = (ArgType) ((IndexInsnNode) insn).getIndex();
277 				addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, clsType));
278 				break;
279 
280 			case CONSTRUCTOR:
281 				ArgType ctrClsType = replaceAnonymousType((ConstructorInsn) insn);
282 				addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, ctrClsType));
283 				break;
284 
285 			case CONST:
286 				LiteralArg constLit = (LiteralArg) insn.getArg(0);
287 				addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, constLit.getType()));
288 				break;
289 
290 			case MOVE_EXCEPTION:
291 				ExcHandlerAttr excHandlerAttr = insn.get(AType.EXC_HANDLER);
292 				if (excHandlerAttr != null) {
293 					for (ClassInfo catchType : excHandlerAttr.getHandler().getCatchTypes()) {
294 						addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, catchType.getType()));
295 					}
296 				} else {
297 					addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, insn.getResult().getInitType()));
298 				}
299 				break;
300 
301 			case INVOKE:
302 				addBound(typeInfo, makeAssignInvokeBound((InvokeNode) insn));
303 				break;
304 
305 			case IGET:
306 				addBound(typeInfo, makeAssignFieldGetBound((IndexInsnNode) insn));
307 				break;
308 
309 			case CHECK_CAST:
310 				addBound(typeInfo, new TypeBoundCheckCastAssign(root, (IndexInsnNode) insn));
311 				break;
312 
313 			default:
314 				ArgType type = insn.getResult().getInitType();
315 				addBound(typeInfo, new TypeBoundConst(BoundEnum.ASSIGN, type));
316 				break;
317 		}
318 	}

420 		if (type.isArray()) {

422 			for (ArgType arrElemType : makePossibleTypesList(type.getArrayElement(), null)) {
423 				list.add(ArgType.array(arrElemType));
424 			}
425 			return list;

428 			for (ITypeBound b : var.getTypeInfo().getBounds()) {
429 				ArgType boundType = b.getType();

456 	private boolean deduceType(MethodNode mth, SSAVar var) {
457 		if (var.isTypeImmutable()) {
458 			return false;
459 		}
460 		ArgType type = var.getTypeInfo().getType();
461 		if (type.isTypeKnown()) {
462 			return false;
463 		}
464 		// try best type from bounds again
465 		if (setBestType(mth, var)) {
466 			return true;
467 		}
468 		// try all possible types (useful for primitives)
469 		if (tryPossibleTypes(mth, var, type)) {
470 			return true;
471 		}
472 		// for objects try super types
473 		if (tryWiderObjects(mth, var)) {
474 			return true;
475 		}
476 		return false;
477 	}

492 	private boolean tryRawType(MethodNode mth, SSAVar var) {
493 		Set<ArgType> objTypes = new LinkedHashSet<>();
494 		for (ITypeBound bound : var.getTypeInfo().getBounds()) {
495 			ArgType boundType = bound.getType();
496 			if (boundType.isTypeKnown() && boundType.isObject()) {
497 				objTypes.add(boundType);
498 			}
499 		}
500 		if (objTypes.isEmpty()) {
501 			return false;
502 		}
503 		for (ArgType objType : objTypes) {
504 			if (checkRawType(mth, var, objType)) {
505 				mth.addDebugComment("Type inference failed for " + var.toShortString() + "."
506 						+ " Raw type applied. Possible types: " + Utils.listToString(objTypes));
507 				return true;
508 			}
509 		}
510 		return false;
511 	}

545 	private int tryInsertVarCast(MethodNode mth, SSAVar var) {
546 		for (ITypeBound bound : var.getTypeInfo().getBounds()) {
547 			ArgType boundType = bound.getType();
548 			if (boundType.isTypeKnown()
549 					&& !boundType.equals(var.getTypeInfo().getType())
550 					&& boundType.containsTypeVariable()
551 					&& !root.getTypeUtils().containsUnknownTypeVar(mth, boundType)) {
552 				if (insertAssignCast(mth, var, boundType)) {
553 					return 1;
554 				}
555 				return insertUseCasts(mth, var);
556 			}
557 		}
558 		return 0;
559 	}

636 	private boolean checkAndSplitConstInsn(MethodNode mth, SSAVar var) {
637 		ArgType type = var.getTypeInfo().getType();
638 		if (type.isTypeKnown() || var.isTypeImmutable()) {
639 			return false;
640 		}
641 		if (var.getUsedInPhi().size() < 2) {
642 			return false;
643 		}
644 		InsnNode assignInsn = var.getAssign().getAssignInsn();
645 		InsnNode constInsn = InsnUtils.checkInsnType(assignInsn, InsnType.CONST);
646 		if (constInsn == null) {
647 			return false;
648 		}
649 		BlockNode blockNode = BlockUtils.getBlockByInsn(mth, constInsn);
650 		if (blockNode == null) {
651 			return false;
652 		}
653 		// for every PHI make separate CONST insn
654 		boolean first = true;
655 		for (PhiInsn phiInsn : var.getUsedInPhi()) {
656 			if (first) {
657 				first = false;
658 				continue;
659 			}
660 			InsnNode copyInsn = constInsn.copyWithNewSsaVar(mth);
661 			copyInsn.add(AFlag.SYNTHETIC);
662 			BlockUtils.insertAfterInsn(blockNode, constInsn, copyInsn);
663 
664 			RegisterArg phiArg = phiInsn.getArgBySsaVar(var);
665 			phiInsn.replaceArg(phiArg, copyInsn.getResult().duplicate());
666 		}
667 		return true;
668 	}

699 		ArgType phiType = getCommonTypeForPhiArgs(phiInsn);

705 		if (insertMovesForPhi(mth, phiInsn, false) == 0) {

726 	private int insertMovesForPhi(MethodNode mth, PhiInsn phiInsn, boolean apply) {
727 		int argsCount = phiInsn.getArgsCount();
728 		int count = 0;
729 		for (int argIndex = 0; argIndex < argsCount; argIndex++) {
730 			RegisterArg reg = phiInsn.getArg(argIndex);
731 			BlockNode startBlock = phiInsn.getBlockByArgIndex(argIndex);
732 			BlockNode blockNode = checkBlockForInsnInsert(startBlock);
733 			if (blockNode == null) {
734 				mth.addDebugComment("Failed to insert an additional move for type inference into block " + startBlock);
735 				return 0;
736 			}
737 			boolean add = true;
738 			SSAVar var = reg.getSVar();
739 			InsnNode assignInsn = var.getAssign().getAssignInsn();
740 			if (assignInsn != null) {
741 				InsnType assignType = assignInsn.getType();
742 				if (assignType == InsnType.CONST
743 						|| (assignType == InsnType.MOVE && var.getUseCount() == 1)) {
744 					add = false;
745 				}
746 			}
747 			if (add) {
748 				count++;
749 				if (apply) {
750 					insertMove(mth, blockNode, phiInsn, reg);
751 				}
752 			}
753 		}
754 		return count;
755 	}

757 	private void insertMove(MethodNode mth, BlockNode blockNode, PhiInsn phiInsn, RegisterArg reg) {
758 		SSAVar var = reg.getSVar();
759 		int regNum = reg.getRegNum();
760 		RegisterArg resultArg = reg.duplicate(regNum, null);
761 		SSAVar newSsaVar = mth.makeNewSVar(resultArg);
762 		RegisterArg arg = reg.duplicate(regNum, var);
763 
764 		InsnNode moveInsn = new InsnNode(InsnType.MOVE, 1);
765 		moveInsn.setResult(resultArg);
766 		moveInsn.addArg(arg);
767 		moveInsn.add(AFlag.SYNTHETIC);
768 		blockNode.getInstructions().add(moveInsn);
769

771 	}

834 		TypeInfo typeInfo = var.getTypeInfo();
835 		if (typeInfo.getType().isTypeKnown()) {
836 			return false;
837 		}

839 			ArgType boundType = bound.getType();
840 			switch (bound.getBound()) {
841 				case ASSIGN:

844 					}

846 				case USE:

849 					}

851 			}
852 		}

855 		for (ITypeBound bound : typeInfo.getBounds()) {
856 			if (bound.getBound() == BoundEnum.USE
857 					&& fixBooleanUsage(mth, bound)) {

864 	private boolean fixBooleanUsage(MethodNode mth, ITypeBound bound) {
865 		ArgType boundType = bound.getType();
866 		if (boundType == ArgType.BOOLEAN
867 				|| (boundType.isTypeKnown() && !boundType.isPrimitive())) {
868 			return false;
869 		}
870 		RegisterArg boundArg = bound.getArg();
871 		if (boundArg == null) {
872 			return false;
873 		}
874 		InsnNode insn = boundArg.getParentInsn();
875 		if (insn == null || insn.getType() == InsnType.IF) {
876 			return false;
877 		}
878 		BlockNode blockNode = BlockUtils.getBlockByInsn(mth, insn);
879 		if (blockNode == null) {
880 			return false;
881 		}
882 		List<InsnNode> insnList = blockNode.getInstructions();
883 		int insnIndex = InsnList.getIndex(insnList, insn);
884 		if (insnIndex == -1) {
885 			return false;
886 		}
887 		InsnType insnType = insn.getType();
888 		if (insnType == InsnType.CAST) {
889 			// replace cast
890 			ArgType type = (ArgType) ((IndexInsnNode) insn).getIndex();
891 			TernaryInsn convertInsn = prepareBooleanConvertInsn(insn.getResult(), boundArg, type);
892 			BlockUtils.replaceInsn(mth, blockNode, insnIndex, convertInsn);
893 			return true;
894 		}
895 		if (insnType == InsnType.ARITH) {
896 			ArithNode arithInsn = (ArithNode) insn;
897 			if (arithInsn.getOp() == ArithOp.XOR && arithInsn.getArgsCount() == 2) {
898 				// replace (boolean ^ 1) with (!boolean)
899 				InsnArg secondArg = arithInsn.getArg(1);
900 				if (secondArg.isLiteral() && ((LiteralArg) secondArg).getLiteral() == 1) {
901 					InsnNode convertInsn = notBooleanToInt(arithInsn, boundArg);
902 					BlockUtils.replaceInsn(mth, blockNode, insnIndex, convertInsn);
903 					return true;
904 				}
905 			}
906 		}
907 
908 		// insert before insn
909 		RegisterArg resultArg = boundArg.duplicateWithNewSSAVar(mth);
910 		TernaryInsn convertInsn = prepareBooleanConvertInsn(resultArg, boundArg, boundType);
911 		insnList.add(insnIndex, convertInsn);
912 		insn.replaceArg(boundArg, convertInsn.getResult().duplicate());
913 		return true;
914 	}
