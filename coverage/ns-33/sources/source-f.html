77 		if (candidateType == null || !candidateType.isTypeKnown()) {
78 			return REJECT;
79 		}
80 
81 		TypeUpdateInfo updateInfo = new TypeUpdateInfo(mth, flags);
82 		TypeUpdateResult result = updateTypeChecked(updateInfo, ssaVar.getAssign(), candidateType);
83 		if (result == REJECT) {
84 			return result;
85 		}
86 		List<TypeUpdateEntry> updates = updateInfo.getUpdates();
87 		if (updates.isEmpty()) {
88 			return SAME;
89 		}
90 		if (Consts.DEBUG_TYPE_INFERENCE) {
91 			LOG.debug("Applying type {} to {}", ssaVar.toShortString(), candidateType);
92 			updates.forEach(updateEntry -> LOG.debug("  {} -> {} in {}",
93 					updateEntry.getType(), updateEntry.getArg(), updateEntry.getArg().getParentInsn()));
94 		}
95 		updateInfo.applyUpdates();

99 	private TypeUpdateResult updateTypeChecked(TypeUpdateInfo updateInfo, InsnArg arg, ArgType candidateType) {
100 		if (candidateType == null) {
101 			throw new JadxRuntimeException("Null type update for arg: " + arg);
102 		}
103 		ArgType currentType = arg.getType();
104 		if (Objects.equals(currentType, candidateType)) {
105 			if (!updateInfo.getFlags().isIgnoreSame()) {
106 				return SAME;
107 			}
108 		} else {
109 			if (candidateType.isWildcard()) {
110 				if (Consts.DEBUG_TYPE_INFERENCE) {
111 					LOG.debug("Wildcard type rejected for {}: candidate={}, current={}", arg, candidateType, currentType);
112 				}
113 				return REJECT;
114 			}
115 
116 			TypeCompareEnum compareResult = comparator.compareTypes(candidateType, currentType);
117 			if (compareResult == TypeCompareEnum.UNKNOWN && updateInfo.getFlags().isIgnoreUnknown()) {
118 				return REJECT;
119 			}
120 			if (arg.isTypeImmutable() && currentType != ArgType.UNKNOWN) {
121 				// don't changed type
122 				if (compareResult == TypeCompareEnum.EQUAL) {
123 					return SAME;
124 				}
125 				if (Consts.DEBUG_TYPE_INFERENCE) {
126 					LOG.debug("Type rejected for {} due to conflict: candidate={}, current={}", arg, candidateType, currentType);
127 				}
128 				return REJECT;
129 			}
130 			if (compareResult.isWider() && !updateInfo.getFlags().isAllowWider()) {
131 				if (Consts.DEBUG_TYPE_INFERENCE) {
132 					LOG.debug("Type rejected for {}: candidate={} is wider than current={}", arg, candidateType, currentType);
133 				}
134 				return REJECT;
135 			}
136 			if (candidateType.containsTypeVariable()) {
137 				// reject unknown type vars
138 				ArgType unknownTypeVar = root.getTypeUtils().checkForUnknownTypeVars(updateInfo.getMth(), candidateType);
139 				if (unknownTypeVar != null) {
140 					if (Consts.DEBUG_TYPE_INFERENCE) {
141 						LOG.debug("Type rejected for {}: candidate: '{}' has unknown type var: '{}'", arg, candidateType, unknownTypeVar);
142 					}
143 					return REJECT;
144 				}
145 			}
146 		}
147 		if (arg instanceof RegisterArg) {
148 			RegisterArg reg = (RegisterArg) arg;
149 			return updateTypeForSsaVar(updateInfo, reg.getSVar(), candidateType);
150 		}
151 		return requestUpdate(updateInfo, arg, candidateType);
152 	}

189 	private TypeUpdateResult requestUpdate(TypeUpdateInfo updateInfo, InsnArg arg, ArgType candidateType) {
190 		if (updateInfo.isProcessed(arg)) {
191 			return CHANGED;
192 		}
193 		updateInfo.requestUpdate(arg, candidateType);
194 		updateInfo.checkUpdatesCount();
195 		try {
196 			TypeUpdateResult result = runListeners(updateInfo, arg, candidateType);
197 			if (result == REJECT) {
198 				updateInfo.rollbackUpdate(arg);
199 			}
200 			return result;
201 		} catch (StackOverflowError | BootstrapMethodError error) {
202 			throw new JadxOverflowException("Type update terminated with stack overflow, arg: " + arg);

246 	private boolean checkBound(ArgType candidateType, ITypeBound bound, ArgType boundType) {
247 		TypeCompareEnum compareResult = comparator.compareTypes(candidateType, boundType);
248 		switch (compareResult) {
249 			case EQUAL:
250 				return true;
251 
252 			case WIDER:
253 				return bound.getBound() != BoundEnum.USE;
254 
255 			case NARROW:
256 				if (bound.getBound() == BoundEnum.ASSIGN) {
257 					return !boundType.isTypeKnown() && checkAssignForUnknown(boundType, candidateType);
258 				}
259 				return true;
260 
261 			case WIDER_BY_GENERIC:
262 			case NARROW_BY_GENERIC:
263 				// allow replace object to same object with known generic type
264 				// due to incomplete information about external methods and fields
265 				return true;
266 
267 			case CONFLICT:
268 				return false;
269 
270 			case UNKNOWN:
271 				LOG.warn("Can't compare types, unknown hierarchy: {} and {}", candidateType, boundType);
272 				comparator.compareTypes(candidateType, boundType);
273 				return true;
274 
275 			default:
276 				throw new JadxRuntimeException("Not processed type compare enum: " + compareResult);
277 		}
278 	}

281 		if (boundType == ArgType.UNKNOWN) {
282 			return true;
283 		}
284 		boolean candidateArray = candidateType.isArray();
285 		if (boundType.isArray() && candidateArray) {
286 			return checkAssignForUnknown(boundType.getArrayElement(), candidateType.getArrayElement());
287 		}
288 		if (candidateArray && boundType.contains(PrimitiveType.ARRAY)) {
289 			return true;
290 		}
291 		if (candidateType.isObject() && boundType.contains(PrimitiveType.OBJECT)) {
292 			return true;
293 		}
294 		if (candidateType.isPrimitive() && boundType.contains(candidateType.getPrimitiveType())) {
295 			return true;
296 		}
297 		return false;
298 	}

348 			Set<ArgType> knownTypeVars, Supplier<ArgType> getReturnType, Function<Integer, ArgType> getArgType) {

350 		RegisterArg resultArg = invoke.getResult();
351 		if (resultArg != null && !resultArg.isTypeImmutable()) {
352 			ArgType returnType = checkType(knownTypeVars, getReturnType.get());
353 			if (returnType != null) {
354 				TypeUpdateResult result = updateTypeChecked(updateInfo, resultArg, returnType);
355 				if (result == REJECT) {
356 					TypeCompareEnum compare = comparator.compareTypes(returnType, resultArg.getType());
357 					if (compare.isWider()) {
358 						return REJECT;
359 					}
360 				}
361 				if (result == CHANGED) {
362 					allSame = false;
363 				}
364 			}
365 		}

371 				if (argType != null) {

373 					if (result == REJECT) {
374 						TypeCompareEnum compare = comparator.compareTypes(argType, invokeArg.getType());
375 						if (compare.isNarrow()) {
376 							return REJECT;
377 						}
378 					}
379 					if (result == CHANGED) {
380 						allSame = false;
381 					}
382 				}
383 			}
384 		}

542 					TypeCompareEnum compResult = comparator.compareTypes(arrayElement, putType);
543 					if (compResult == TypeCompareEnum.WIDER || compResult == TypeCompareEnum.WIDER_BY_GENERIC) {
544 						// allow wider result (i.e allow put in Object[] any objects)
545 						return CHANGED;
546 					}
547 				}
548 			}
549 			return result;
550 		}
551 		if (arrArg == putArg) {
552 			return updateTypeChecked(updateInfo, arrArg, ArgType.array(candidateType));
553 		}
554 		// index
555 		return SAME;
556 	}

558 	private TypeUpdateResult ifListener(TypeUpdateInfo updateInfo, InsnNode insn, InsnArg arg, ArgType candidateType) {
559 		InsnArg firstArg = insn.getArg(0);
560 		InsnArg secondArg = insn.getArg(1);
561 		InsnArg updateArg = firstArg == arg ? secondArg : firstArg;
562 		TypeUpdateResult result = updateTypeChecked(updateInfo, updateArg, candidateType);
563 		if (result == REJECT) {
564 			// soft checks for objects and array - exact type not compared
565 			ArgType updateArgType = updateArg.getType();
566 			if (candidateType.isObject() && updateArgType.canBeObject()) {
567 				return SAME;
568 			}
569 			if (candidateType.isArray() && updateArgType.canBeArray()) {
570 				return SAME;
571 			}
572 			if (candidateType.isPrimitive()) {
573 				if (updateArgType.canBePrimitive(candidateType.getPrimitiveType())) {
574 					return SAME;
575 				}
576 				if (updateArgType.isTypeKnown() && candidateType.getRegCount() == updateArgType.getRegCount()) {
577 					return SAME;
578 				}
579 			}
580 		}
581 		return result;
582 	}
