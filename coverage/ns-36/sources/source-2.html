412 		visitor.accept(start);
413 		if (start == end) {
414 			return true;
415 		}
416 		if (start.getCleanSuccessors().contains(end)) {
417 			visitor.accept(end);
418 			return true;
419 		}
420 		// DFS on clean successors
421 		BitSet visited = newBlocksBitSet(mth);
422 		Deque<BlockNode> queue = new ArrayDeque<>();
423 		queue.addLast(start);
424 		while (true) {
425 			BlockNode current = queue.peekLast();
426 			if (current == null) {
427 				return false;
428 			}
429 			boolean added = false;
430 			for (BlockNode next : current.getCleanSuccessors()) {
431 				if (next == end) {
432 					queue.removeFirst(); // start already visited
433 					queue.addLast(next);
434 					queue.forEach(visitor);
435 					return true;
436 				}
437 				int id = next.getId();
438 				if (!visited.get(id)) {
439 					visited.set(id);
440 					queue.addLast(next);
441 					added = true;
442 					break;
443 				}
444 			}
445 			if (!added) {
446 				queue.pollLast();
447 				if (queue.isEmpty()) {
448 					return false;
449 				}
450 			}

672 	/**
673 	 * Return common cross block for input set.
674 	 *
675 	 * @return null if cross is a method exit block.
676 	 */
677 	@Nullable
678 	public static BlockNode getPathCross(MethodNode mth, Collection<BlockNode> blocks) {
679 		BitSet domFrontBS = newBlocksBitSet(mth);
680 		boolean first = true;
681 		for (BlockNode b : blocks) {
682 			if (first) {
683 				domFrontBS.or(b.getDomFrontier());
684 				first = false;
685 			} else {
686 				domFrontBS.and(b.getDomFrontier());
687 			}
688 		}
689 		domFrontBS.clear(mth.getExitBlock().getId());
690 		if (domFrontBS.isEmpty()) {
691 			return null;
692 		}
693 		BlockNode oneBlock = bitSetToOneBlock(mth, domFrontBS);
694 		if (oneBlock != null) {
695 			return oneBlock;
696 		}
697 		BitSet excluded = newBlocksBitSet(mth);
698 		// exclude method exit and loop start blocks
699 		excluded.set(mth.getExitBlock().getId());
700 		// exclude loop start blocks
701 		mth.getLoops().forEach(l -> excluded.set(l.getStart().getId()));
702 		if (!mth.isNoExceptionHandlers()) {
703 			// exclude exception handlers paths
704 			mth.getExceptionHandlers().forEach(h -> excluded.or(h.getHandlerBlock().getDomFrontier()));
705 		}
706 		domFrontBS.andNot(excluded);
707 		oneBlock = bitSetToOneBlock(mth, domFrontBS);
708 		if (oneBlock != null) {
709 			return oneBlock;
710 		}
711 		BitSet combinedDF = newBlocksBitSet(mth);
712 		int k = mth.getBasicBlocks().size();
713 		while (true) {
714 			// collect dom frontier blocks from current set until only one block left
715 			forEachBlockFromBitSet(mth, domFrontBS, block -> {
716 				BitSet domFrontier = block.getDomFrontier();
717 				if (!domFrontier.isEmpty()) {
718 					combinedDF.or(domFrontier);
719 					combinedDF.clear(block.getId());
720 				}
721 			});
722 			combinedDF.andNot(excluded);
723 			int cardinality = combinedDF.cardinality();
724 			if (cardinality == 1) {
725 				return bitSetToOneBlock(mth, combinedDF);
726 			}
727 			if (cardinality == 0) {
728 				return null;
729 			}
730 			if (k-- < 0) {
731 				mth.addWarnComment("Path cross not found for " + blocks + ", limit reached: " + mth.getBasicBlocks().size());
732 				return null;
733 			}
734 			// replace domFrontBS with combinedDF
735 			domFrontBS.clear();
736 			domFrontBS.or(combinedDF);
737 			combinedDF.clear();
738 		}
739 	}
