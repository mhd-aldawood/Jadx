78 	/**
79 	 * The general contract of DataInput doesn't guarantee all the bytes requested will be skipped
80 	 * and failure can occur for many reasons. We override this to try harder to skip all the bytes
81 	 * requested (this is similar to DataInputStream's wrapper).
82 	 */
83 	@Override
84 	@SuppressWarnings("InnerAssignment")
85 	public final int skipBytes(int n) throws IOException {
86 		int total = 0;
87 		int cur;
88 
89 		while ((total < n) && ((cur = super.skipBytes(n - total)) > 0)) {
90 			total += cur;
91 		}
92 		return total;
93 	}
