99 	void decode() throws IOException {
100 		int size = is.readInt32();
101 		while (is.getPos() < size) {
102 			int type = is.readInt16();
103 			switch (type) {
104 				case RES_NULL_TYPE:
105 					// NullType is just doing nothing
106 					break;
107 				case RES_STRING_POOL_TYPE:
108 					strings = parseStringPoolNoType();
109 					valuesParser = new ValuesParser(strings, resNames);
110 					break;
111 				case RES_XML_RESOURCE_MAP_TYPE:
112 					parseResourceMap();
113 					break;
114 				case RES_XML_START_NAMESPACE_TYPE:
115 					parseNameSpace();
116 					break;
117 				case RES_XML_CDATA_TYPE:
118 					parseCData();
119 					break;
120 				case RES_XML_END_NAMESPACE_TYPE:
121 					parseNameSpaceEnd();
122 					break;
123 				case RES_XML_START_ELEMENT_TYPE:
124 					parseElement();
125 					break;
126 				case RES_XML_END_ELEMENT_TYPE:
127 					parseElementEnd();
128 					break;
129 
130 				default:
131 					if (namespaceDepth == 0) {
132 						// skip padding on file end
133 						return;
134 					}
135 					die("Type: 0x" + Integer.toHexString(type) + " not yet implemented");
136 					break;
137 			}
138 		}
139 	}
