398 	private short getNaturalSdkVersionRequirement() {
399 		if ((uiMode & MASK_UI_MODE_TYPE) == UI_MODE_TYPE_VR_HEADSET || (colorMode & COLOR_WIDE_MASK) != 0
400 				|| ((colorMode & COLOR_HDR_MASK) != 0)) {
401 			return SDK_OREO;
402 		}
403 		if ((screenLayout2 & MASK_SCREENROUND) != 0) {
404 			return SDK_MNC;
405 		}
406 		if (density == DENSITY_ANY) {
407 			return SDK_LOLLIPOP;
408 		}
409 		if (smallestScreenWidthDp != 0 || screenWidthDp != 0 || screenHeightDp != 0) {
410 			return SDK_HONEYCOMB_MR2;
411 		}
412 		if ((uiMode & (MASK_UI_MODE_TYPE | MASK_UI_MODE_NIGHT)) != UI_MODE_NIGHT_ANY) {
413 			return SDK_FROYO;
414 		}
415 		if ((screenLayout & (MASK_SCREENSIZE | MASK_SCREENLONG)) != SCREENSIZE_ANY || density != DENSITY_DEFAULT) {
416 			return SDK_DONUT;
417 		}
418 		return 0;
419 	}

421 	private String getLocaleString() {
422 		StringBuilder sb = new StringBuilder();
423 
424 		// check for old style non BCP47 tags
425 		// allows values-xx-rXX, values-xx, values-xxx-rXX
426 		// denies values-xxx, anything else
427 		if (localeVariant == null && localeScript == null && (region[0] != '\00' || language[0] != '\00')
428 				&& region.length != 3) {
429 			sb.append('-').append(language);
430 			if (region[0] != '\00') {
431 				sb.append("-r").append(region);
432 			}
433 		} else { // BCP47
434 			if (language[0] == '\00' && region[0] == '\00') {
435 				return sb.toString(); // early return, no language or region
436 			}
437 			sb.append("-b+");
438 			if (language[0] != '\00') {
439 				sb.append(language);
440 			}
441 			if (localeScript != null && localeScript.length == 4) {
442 				sb.append('+').append(localeScript);
443 			}
444 			if ((region.length == 2 || region.length == 3) && region[0] != '\00') {
445 				sb.append('+').append(region);
446 			}
447 			if (localeVariant != null && localeVariant.length >= 5) {
448 				sb.append('+').append(toUpper(localeVariant));
449 			}
450 		}
451 		return sb.toString();
452 	}
