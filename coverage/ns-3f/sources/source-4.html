449 		if (type == null) {

480 			}

725 	private boolean decodeRuntimeValue(RuntimeValueTreeNode valNode) {
726 		RuntimeValue rValue = valNode.getRuntimeValue();
727 		RuntimeType type = rValue.getType();
728 		if (!valNode.isAbsoluteType()) {
729 			valNode.updateType(null);
730 		}
731 		try {
732 			switch (type) {
733 				case OBJECT:
734 					return decodeObject(valNode);
735 				case STRING:
736 					String str = "\"" + debugger.readStringSync(rValue) + "\"";
737 					valNode.updateType("java.lang.String")
738 							.updateTypeID(debugger.readID(rValue))
739 							.updateValue(str);
740 					break;
741 				case INT:
742 					valNode.updateValue(Integer.toString(debugger.readInt(rValue)));
743 					break;
744 				case LONG:
745 					valNode.updateValue(Long.toString(debugger.readAll(rValue)));
746 					break;
747 				case ARRAY:
748 					decodeArrayVal(valNode);
749 					break;
750 				case BOOLEAN: {
751 					int b = debugger.readByte(rValue);
752 					valNode.updateValue(b == 1 ? "true" : "false");
753 					break;
754 				}
755 				case SHORT:
756 					valNode.updateValue(Short.toString(debugger.readShort(rValue)));
757 					break;
758 				case CHAR:
759 				case BYTE: {
760 					int b = (int) debugger.readAll(rValue);
761 					if (DbgUtils.isPrintableChar(b)) {
762 						valNode.updateValue(type == RuntimeType.CHAR ? String.valueOf((char) b) : String.valueOf((byte) b));
763 					} else {
764 						valNode.updateValue(String.valueOf(b));
765 					}
766 					break;
767 				}
768 				case DOUBLE:
769 					double d = debugger.readDouble(rValue);
770 					valNode.updateValue(Double.toString(d));
771 					break;
772 				case FLOAT:
773 					float f = debugger.readFloat(rValue);
774 					valNode.updateValue(Float.toString(f));
775 					break;
776 				case VOID:
777 					valNode.updateType("void");
778 					break;
779 				case THREAD:
780 					valNode.updateType("thread").updateTypeID(debugger.readID(rValue));
781 					break;
782 				case THREAD_GROUP:
783 					valNode.updateType("thread_group").updateTypeID(debugger.readID(rValue));
784 					break;
785 				case CLASS_LOADER:
786 					valNode.updateType("class_loader").updateTypeID(debugger.readID(rValue));
787 					break;
788 				case CLASS_OBJECT:
789 					valNode.updateType("class_object").updateTypeID(debugger.readID(rValue));
790 					break;
791 			}
792 		} catch (SmaliDebuggerException e) {
793 			logErr(e);
794 			return false;
795 		}
796 		return true;
797 	}

799 	private boolean decodeObject(RuntimeValueTreeNode valNode) {
800 		RuntimeValue rValue = valNode.getRuntimeValue();
801 		boolean ok = true;
802 		if (debugger.readID(rValue) == 0) {
803 			if (valNode.isAbsoluteType()) {
804 				valNode.updateValue("null");
805 				return ok;
806 			} else if (!art.readNullObject()) {
807 				valNode.updateType(art.typeForNull());
808 				valNode.updateValue("0");
809 				return ok;
810 			}
811 		}
812 		String sig;
813 		try {
814 			sig = debugger.readObjectSignatureSync(rValue);
815 			valNode.updateType(String.format("%s@%d", DbgUtils.classSigToRawFullName(sig),
816 					debugger.readID(rValue)));
817 		} catch (SmaliDebuggerException e) {
818 			ok = debugger.errIsInvalidObject(e.getErrCode()) && valNode instanceof RegTreeNode;
819 			if (ok) {
820 				try {
821 					RegTreeNode reg = (RegTreeNode) valNode;
822 					RuntimeRegister rr = debugger.getRegisterSync(
823 							cur.frame.getThreadID(),
824 							cur.frame.getFrame().getID(),
825 							reg.getRuntimeRegNum(), RuntimeType.INT);
826 					reg.updateReg(rr);
827 					rValue = rr;
828 					valNode.updateType(RuntimeType.INT.getDesc());
829 					valNode.updateValue(Long.toString((int) debugger.readAll(rValue)));
830 				} catch (SmaliDebuggerException except) {
831 					logErr(except, String.format("Update %s failed, %s", valNode.getName(), except.getMessage()));
832 					valNode.updateValue(except.getMessage());
833 					ok = false;
834 				}
835 			} else {
836 				logErr(e);
837 			}
838 		}
839 		return ok;
840 	}

842 	private void decodeArrayVal(RuntimeValueTreeNode valNode) throws SmaliDebuggerException {
843 		String type = debugger.readObjectSignatureSync(valNode.getRuntimeValue());
844 		ArgType argType = ArgType.parse(type);
845 		String javaType = argType.toString();
846 		Entry<Integer, List<Long>> ret = debugger.readArray(valNode.getRuntimeValue(), 0, 0);
847 		javaType = javaType.substring(0, javaType.length() - 1) + ret.getKey() + "]";
848 		valNode.updateType(javaType + "@" + debugger.readID(valNode.getRuntimeValue()));
849 
850 		if (argType.getArrayElement().isPrimitive()) {
851 			for (Long aLong : ret.getValue()) {
852 				valNode.add(new DefaultMutableTreeNode(Long.toString(aLong)));
853 			}
854 			return;
855 		}
856 		String typeSig = type.substring(1);
857 		if (DbgUtils.isStringObjectSig(typeSig)) {
858 			for (Long aLong : ret.getValue()) {
859 				valNode.add(new DefaultMutableTreeNode(debugger.readStringSync(aLong)));
860 			}
861 			return;
862 		}
863 		typeSig = DbgUtils.classSigToRawFullName(typeSig);
864 		for (Long aLong : ret.getValue()) {
865 			valNode.add(new DefaultMutableTreeNode(String.format("%s@%d", typeSig, aLong)));
866 		}
867 	}
