672 	 * Read & decode packets from Socket connection
673 	 */
674 	private void decodingLoop() {
675 		Executors.newSingleThreadExecutor().execute(() -> {
676 			boolean errFromCallback;
677 			for (;;) {
678 				errFromCallback = false;
679 				try {
680 					Packet res = readPacket(inputStream);
681 					suspendInfo.nextRound();
682 					ICommandResult callback = callbackMap.remove(res.getID());
683 					if (callback != null) {
684 						if (callback != SKIP_RESULT) {
685 							errFromCallback = true;
686 							callback.onCommandReply(res);
687 						}
688 						continue;
689 					}
690 					if (res.getCommandSetID() == 64 && res.getCommandID() == 100) { // command from JVM
691 						errFromCallback = true;
692 						decodeCompositeEvents(res);
693 					} else {
694 						printUnexpectedID(res.getID());
695 					}
696 				} catch (SmaliDebuggerException e) {
697 					LOG.error("Error in debugger decoding loop", e);
698 					if (!errFromCallback) { // fatal error
699 						break;
700 					}
701 				}
702 			}
703 			suspendInfo.setTerminated();
704 			clearWaitingSyncQueue();
705 			suspendListener.onSuspendEvent(suspendInfo);
706 		});
707 	}

721 	 * It should be used in a thread.
722 	 */
723 	private Packet sendCommandSync(ByteBuffer buf) throws SmaliDebuggerException {
724 		SynchronousQueue<Packet> store = syncQueuePool.get();
725 		sendCommand(buf, res -> {
726 			try {
727 				store.put(res);
728 			} catch (Exception e) {
729 				LOG.error("Command send failed", e);
730 			}
731 		});
732 		Integer id = syncQueueID.getAndAdd(1);
733 		try {
734 			syncQueueMap.put(id, Thread.currentThread());
735 			return store.take();
736 		} catch (InterruptedException e) {
737 			throw new SmaliDebuggerException(e);
738 		} finally {
739 			syncQueueMap.remove(id);
740 			syncQueuePool.put(store);
741 		}
742 	}

759 	private void decodeCompositeEvents(Packet res) throws SmaliDebuggerException {
760 		EventData data = jdwp.event().cmdComposite().decode(res.getBuf(), JDWP.PACKET_HEADER_SIZE);
761 		for (JDWP.EventRequestDecoder event : data.events) {
762 			EventListenerAdapter listener = eventListenerMap.get(event.getRequestID());
763 			if (listener == null) {
764 				LOG.error("Missing handler for id: {}", event.getRequestID());
765 				continue;
766 			}
767 			if (event instanceof VMStartEvent) {
768 				listener.onVMStart((VMStartEvent) event);
769 				return;
770 			}
771 			if (event instanceof VMDeathEvent) {
772 				listener.onVMDeath((VMDeathEvent) event);
773 				return;
774 			}
775 			if (event instanceof SingleStepEvent) {
776 				listener.onSingleStep((SingleStepEvent) event);
777 				return;
778 			}
779 			if (event instanceof BreakpointEvent) {
780 				listener.onBreakpoint((BreakpointEvent) event);
781 				return;
782 			}
783 			if (event instanceof MethodEntryEvent) {
784 				listener.onMethodEntry((MethodEntryEvent) event);
785 				return;
786 			}
787 			if (event instanceof MethodExitEvent) {
788 				listener.onMethodExit((MethodExitEvent) event);
789 				return;
790 			}
791 			if (event instanceof MethodExitWithReturnValueEvent) {
792 				listener.onMethodExitWithReturnValue((MethodExitWithReturnValueEvent) event);
793 				return;
794 			}
795 			if (event instanceof MonitorContendedEnterEvent) {
796 				listener.onMonitorContendedEnter((MonitorContendedEnterEvent) event);
797 				return;
798 			}
799 			if (event instanceof MonitorContendedEnteredEvent) {
800 				listener.onMonitorContendedEntered((MonitorContendedEnteredEvent) event);
801 				return;
802 			}
803 			if (event instanceof MonitorWaitEvent) {
804 				listener.onMonitorWait((MonitorWaitEvent) event);
805 				return;
806 			}
807 			if (event instanceof MonitorWaitedEvent) {
808 				listener.onMonitorWaited((MonitorWaitedEvent) event);
809 				return;
810 			}
811 			if (event instanceof ExceptionEvent) {
812 				listener.onException((ExceptionEvent) event);
813 				return;
814 			}
815 			if (event instanceof ThreadStartEvent) {
816 				listener.onThreadStart((ThreadStartEvent) event);
817 				return;
818 			}
819 			if (event instanceof ThreadDeathEvent) {
820 				listener.onThreadDeath((ThreadDeathEvent) event);
821 				return;
822 			}
823 			if (event instanceof ClassPrepareEvent) {
824 				listener.onClassPrepare((ClassPrepareEvent) event);
825 				return;
826 			}
827 			if (event instanceof ClassUnloadEvent) {
828 				listener.onClassUnload((ClassUnloadEvent) event);
829 				return;
830 			}
831 			if (event instanceof FieldAccessEvent) {
832 				listener.onFieldAccess((FieldAccessEvent) event);
833 				return;
834 			}
835 			if (event instanceof FieldModificationEvent) {
836 				listener.onFieldModification((FieldModificationEvent) event);
837 				return;
838 			}
839 			throw new SmaliDebuggerException("Unexpected event: " + event);
840 		}
841 	}
