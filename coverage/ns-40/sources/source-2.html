356 			} else if (insn.getIndexType() == InsnIndexType.STRING_REF) {

568 	private void writeEncodedValue(SmaliWriter smali, EncodedValue value, boolean wrapArray) {
569 		switch (value.getType()) {
570 			case ENCODED_ARRAY:
571 				smali.add("{");
572 				if (wrapArray) {
573 					smali.incIndent();
574 					smali.startLine();
575 				}
576 				List<EncodedValue> values = (List<EncodedValue>) value.getValue();
577 				for (int i = 0; i < values.size(); i++) {
578 					writeEncodedValue(smali, values.get(i), wrapArray);
579 					if (i != values.size() - 1) {
580 						smali.add(",");
581 						if (wrapArray) {
582 							smali.startLine();
583 						} else {
584 							smali.add(" ");
585 						}
586 					}
587 				}
588 				if (wrapArray) {
589 					smali.decIndent();
590 					smali.startLine("}");
591 				}
592 				break;
593 			case ENCODED_NULL:
594 				smali.add("null");
595 				break;
596 			case ENCODED_ANNOTATION:
597 				writeAnnotation(smali, (IAnnotation) value.getValue());
598 				break;
599 			case ENCODED_BYTE:
600 				smali.add(TypeGen.formatByte((Byte) value.getValue(), false));
601 				break;
602 			case ENCODED_SHORT:
603 				smali.add(TypeGen.formatShort((Short) value.getValue(), false));
604 				break;
605 			case ENCODED_CHAR:
606 				smali.add(smali.getClassNode().root().getStringUtils().unescapeChar((Character) value.getValue()));
607 				break;
608 			case ENCODED_INT:
609 				smali.add(TypeGen.formatInteger((Integer) value.getValue(), false));
610 				break;
611 			case ENCODED_LONG:
612 				smali.add(TypeGen.formatLong((Long) value.getValue(), false));
613 				break;
614 			case ENCODED_FLOAT:
615 				smali.add(TypeGen.formatFloat((Float) value.getValue()));
616 				break;
617 			case ENCODED_DOUBLE:
618 				smali.add(TypeGen.formatDouble((Double) value.getValue()));
619 				break;
620 			case ENCODED_STRING:
621 				smali.add(smali.getClassNode().root().getStringUtils().unescapeString((String) value.getValue()));
622 				break;
623 			case ENCODED_TYPE:
624 				smali.add(ArgType.parse((String) value.getValue()) + ".class");
625 				break;
626 			default:
627 				smali.add(String.valueOf(value.getValue()));
628 		}
629 	}

728 	private boolean fmtPayloadInsn(InsnData insn, LineInfo line) {
729 		Opcode opcode = insn.getOpcode();
730 		if (opcode == PACKED_SWITCH_PAYLOAD) {
731 			line.getLineWriter().append("packed-switch-payload");
732 			line.addInsnLine(insn.getOffset(), line.getLineWriter().toString());
733 
734 			ISwitchPayload payload = (ISwitchPayload) insn.getPayload();
735 			if (payload != null) {
736 				fmtSwitchPayload(insn, FMT_P_SWITCH_CASE, FMT_P_SWITCH_CASE_TAG, line, payload, insn.getOffset());
737 			}
738 			return true;
739 		}
740 		if (opcode == SPARSE_SWITCH_PAYLOAD) {
741 			line.getLineWriter().append("sparse-switch-payload");
742 			line.addInsnLine(insn.getOffset(), line.getLineWriter().toString());
743 
744 			ISwitchPayload payload = (ISwitchPayload) insn.getPayload();
745 			if (payload != null) {
746 				fmtSwitchPayload(insn, FMT_S_SWITCH_CASE, FMT_S_SWITCH_CASE_TAG, line, payload, insn.getOffset());
747 			}
748 			return true;
749 		}
750 		if (opcode == FILL_ARRAY_DATA_PAYLOAD) {
751 			line.getLineWriter().append("fill-array-data-payload");
752 			line.addInsnLine(insn.getOffset(), line.getLineWriter().toString());
753 			return true;
754 		}
755 		return false;
756 	}

758 	private void fmtSwitchPayload(InsnData insn, String fmtTarget, String fmtTag, LineInfo line,
759 			ISwitchPayload payload, int curOffset) {
760 		int lineStart = getInsnColStart();
761 		lineStart += CODE_OFFSET_COLUMN_WIDTH + 1 + 1; // plus 1s for space and the ':'
762 		String basicIndent = new String(new byte[lineStart]).replace("\0", " ");
763 		String indent = SmaliWriter.INDENT_STR + basicIndent;
764 		int[] keys = payload.getKeys();
765 		int[] targets = payload.getTargets();
766 		int opcodeOffset = line.payloadOffsetMap.get(curOffset);
767 		for (int i = 0; i < keys.length; i++) {
768 			int target = opcodeOffset + targets[i];
769 			line.addInsnLine(insn.getOffset(),
770 					String.format("%scase %d: -> " + fmtTarget, indent, keys[i], target));
771 			line.addTip(target,
772 					String.format(fmtTag, target), String.format(" # case %d", keys[i]));
773 		}
774 		line.addInsnLine(insn.getOffset(), basicIndent + ".end payload");
775 	}

972 			Map<String, Object> tip = tipMap.get(codeOffset);
973 			if (tip != null) {
974 				for (Entry<String, Object> entry : tip.entrySet()) {
975 					int start = Math.max(0, lineOffset - entry.getKey().length());
976 					if (start > 0) {
977 						smali.add(new String(new byte[start]).replace("\0", " "));
978 					}

980 						smali.add(String.format("%s # %d refs", entry.getKey(), entry.getValue()))
981 								.startLine();

983 						smali.add(String.format("%s%s", entry.getKey(), entry.getValue()))
984 								.startLine();
985 					} else if (entry.getValue() instanceof List) {
986 						List<String> extras = (List<String>) entry.getValue();
987 						smali.add(String.format("%s%s", entry.getKey(), extras.get(0)))
988 								.startLine();
989 						String pad = new String(new byte[lineOffset]).replace("\0", " ");
990 						for (int i = 1; i < extras.size(); i++) {
991 							smali.add(String.format("%s%s", pad, extras.get(i)))
992 									.startLine();
993 						}
994 					} else {
995 						smali.add(String.format("%s%s", entry.getKey(), entry.getValue()))
996 								.startLine();
997 					}
998 				}
999 			}
