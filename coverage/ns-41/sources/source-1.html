620 		public static byte[] readStdout(InputStream inputStream) throws IOException {
621 			byte[] header = new byte[5];
622 			byte[] payload = new byte[0];
623 			byte[] tempBuf = new byte[0];
624 			for (boolean exit = false; !exit;) {
625 				if (inputStream.read(header, 0, 5) == 5) {
626 					exit = header[0] == ID_EXIT;
627 					int payloadSize = readInt(header, 1);
628 					if (tempBuf.length < payloadSize) {
629 						tempBuf = new byte[payloadSize];
630 					}
631 					int readSize = inputStream.read(tempBuf, 0, payloadSize);
632 					if (readSize != payloadSize) {
633 						return null; // we don't want corrupted data.
634 					}
635 					payload = appendBytes(payload, tempBuf, readSize);
636 				}
