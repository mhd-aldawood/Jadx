281 	public MutableTreeNode resolveMethod(String descr) {
282 		try {
283 			String[] parts = removeQuotes(descr).split(" ", 3);
284 			String cls = Utils.cleanObjectName(parts[0].replace('$', '.'));
285 			String mth = parts[1] + parts[2].replace(" ", "");
286 			MainWindow mainWindow = getTabbedPane().getMainWindow();
287 			JadxWrapper wrapper = mainWindow.getWrapper();
288 			JavaClass javaClass = wrapper.searchJavaClassByRawName(cls);
289 			if (javaClass == null) {
290 				return new TextTreeNode(cls + "." + mth);
291 			}
292 			JavaMethod javaMethod = javaClass.searchMethodByShortId(mth);
293 			if (javaMethod == null) {
294 				return new TextTreeNode(javaClass.getFullName() + "." + mth);
295 			}
296 			return new MethodTreeNode(javaMethod);
297 		} catch (Exception e) {
298 			LOG.error("Failed to parse method descriptor string", e);
299 			return new TextTreeNode(descr);
300 		}
301 	}
