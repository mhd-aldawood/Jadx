584 	private void upgradeSettings(int fromVersion) {
585 		LOG.debug("upgrade settings from version: {} to {}", fromVersion, CURRENT_SETTINGS_VERSION);
586 		if (fromVersion == 0) {
587 			setDeobfuscationMinLength(3);
588 			setDeobfuscationMaxLength(64);
589 			setDeobfuscationUseSourceNameAsAlias(true);
590 			setDeobfuscationParseKotlinMetadata(true);
591 			setDeobfuscationForceSave(false);
592 			setThreadsCount(JadxArgs.DEFAULT_THREADS_COUNT);
593 			setReplaceConsts(true);
594 			setSkipResources(false);
595 			setAutoStartJobs(false);
596 			setAutoSaveProject(true);
597 			fromVersion++;
598 		}
599 		if (fromVersion == 1) {
600 			setEditorThemePath(EditorTheme.getDefaultTheme().getPath());
601 			fromVersion++;
602 		}
603 		if (fromVersion == 2) {
604 			if (getDeobfuscationMinLength() == 4) {
605 				setDeobfuscationMinLength(3);
606 			}
607 			fromVersion++;
608 		}
609 		if (fromVersion == 3) {
610 			setLangLocale(NLS.defaultLocale());
611 			fromVersion++;
612 		}
613 		if (fromVersion == 4) {
614 			setUseImports(true);
615 			fromVersion++;
616 		}
617 		if (fromVersion == 5) {
618 			setRespectBytecodeAccessModifiers(false);
619 			fromVersion++;
620 		}
621 		if (fromVersion == 6) {
622 			if (getFont().getFontName().equals("Hack Regular")) {
623 				setFont(null);
624 			}
625 			fromVersion++;
626 		}
627 		if (fromVersion == 7) {
628 			outDir = null;
629 			outDirSrc = null;
630 			outDirRes = null;
631 			fromVersion++;
632 		}
633 		if (fromVersion == 8) {
634 			fromVersion++;
635 		}
636 		if (fromVersion == 9) {
637 			showHeapUsageBar = false;
638 			fromVersion++;
639 		}
640 		if (fromVersion == 10) {
641 			srhResourceSkipSize = 3;
642 			srhResourceFileExt = ".xml|.html|.js|.json|.txt";
643 			fontStr = fontStr.replace('-', '/');
644 			fromVersion++;
645 		}
646 		if (fromVersion == 11) {
647 			inlineMethods = true;
648 			fromVersion++;
649 		}
650 		if (fromVersion == 12) {
651 			alwaysSelectOpened = false;
652 			fromVersion++;
653 		}
654 		if (fromVersion == 13) {
655 			lafTheme = LafManager.INITIAL_THEME_NAME;
656 			fromVersion++;
657 		}
658 		if (fromVersion == 14) {
659 			useKotlinMethodsForVarNames = JadxArgs.UseKotlinMethodsForVarNames.APPLY;
660 			fromVersion++;
661 		}
662 		if (fromVersion != CURRENT_SETTINGS_VERSION) {
663 			throw new JadxRuntimeException("Incorrect settings upgrade");
664 		}
665 		settingsVersion = CURRENT_SETTINGS_VERSION;
666 		sync();
667 	}
