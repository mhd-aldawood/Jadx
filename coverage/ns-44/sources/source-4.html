663 			jLabel.setLabelFor(comp);
664 			jLabel.setHorizontalAlignment(SwingConstants.LEFT);
665 			c.gridx = 0;

667 			c.anchor = GridBagConstraints.LINE_START;

669 			c.fill = GridBagConstraints.NONE;

693 	private static class JPreferredFontChooser extends JFontChooser {
694 		private static final String[] PREFERRED_FONTS = new String[] {
695 				"Monospaced", "Consolas", "Courier", "Courier New",
696 				"Lucida Sans Typewriter", "Lucida Console",
697 				"SimSun", "SimHei",
698 		};
699 
700 		private String[] filteredFonts;
701 
702 		@Override
703 		protected String[] getFontFamilies() {
704 			if (filteredFonts == null) {
705 				GraphicsEnvironment env = GraphicsEnvironment.getLocalGraphicsEnvironment();
706 				Set<String> fontSet = new HashSet<>();
707 				Collections.addAll(fontSet, env.getAvailableFontFamilyNames());
708 				ArrayList<String> found = new ArrayList<>(PREFERRED_FONTS.length);
709 				for (String font : PREFERRED_FONTS) {
710 					if (fontSet.contains(font)) {
711 						found.add(font);
712 					}
713 				}
714 				if (found.size() == PREFERRED_FONTS.length) {
715 					filteredFonts = PREFERRED_FONTS;
716 				} else if (found.size() > 0) {
717 					filteredFonts = new String[found.size()];
718 					for (int i = 0; i < found.size(); i++) {
719 						filteredFonts[i] = found.get(i);
720 					}
721 				} else {
722 					// this machine is crazy.
723 					LOG.warn("Can't found any preferred fonts for smali, use all available.");
724 					filteredFonts = env.getAvailableFontFamilyNames();
725 				}
726 			}
727 			return filteredFonts;
728 		}
