552 		synchronized (DECOMPILER_TASK_SYNC) {

564 	}

639 	private void expand(TreeNode node, List<String[]> treeExpansions) {
640 		TreeNode[] pathNodes = treeModel.getPathToRoot(node);
641 		if (pathNodes == null) {
642 			return;
643 		}
644 		TreePath path = new TreePath(pathNodes);
645 		for (String[] expansion : treeExpansions) {
646 			if (Arrays.equals(expansion, getPathExpansion(path))) {
647 				tree.expandPath(path);
648 				break;
649 			}
650 		}
651 		for (int i = node.getChildCount() - 1; i >= 0; i--) {
652 			expand(node.getChildAt(i), treeExpansions);
653 		}
654 	}

685 	private boolean nodeClickAction(@Nullable Object obj) {
686 		if (obj == null) {
687 			return false;
688 		}
689 		try {
690 			if (obj instanceof JResource) {
691 				JResource res = (JResource) obj;
692 				ResourceFile resFile = res.getResFile();
693 				if (resFile != null && JResource.isSupportedForView(resFile.getType())) {
694 					return tabbedPane.showNode(res);
695 				}
696 			} else if (obj instanceof JNode) {
697 				JNode node = (JNode) obj;
698 				if (node.getRootClass() != null) {
699 					tabbedPane.codeJump(new JumpPosition(node));
700 					return true;
701 				}
702 				return tabbedPane.showNode(node);
703 			}
704 		} catch (Exception e) {
705 			LOG.error("Content loading error", e);
706 		}
707 		return false;
708 	}

724 	@Nullable
725 	private JNode getJNodeUnderMouse(MouseEvent mouseEvent) {
726 		TreePath path = tree.getClosestPathForLocation(mouseEvent.getX(), mouseEvent.getY());
727 		if (path == null) {
728 			return null;
729 		}
730 		// allow 'closest' path only at the right of the item row
731 		Rectangle pathBounds = tree.getPathBounds(path);
732 		if (pathBounds != null) {
733 			int y = mouseEvent.getY();
734 			if (y < pathBounds.y || y > (pathBounds.y + pathBounds.height)) {
735 				return null;
736 			}
737 			if (mouseEvent.getX() < pathBounds.x) {
738 				// exclude expand/collapse events
739 				return null;
740 			}
741 		}
742 		Object obj = path.getLastPathComponent();
743 		if (obj instanceof JNode) {
744 			tree.setSelectionPath(path);
745 			return (JNode) obj;
746 		}
747 		return null;
748 	}

750 	public void syncWithEditor() {
751 		ContentPanel selectedContentPanel = tabbedPane.getSelectedCodePanel();
752 		if (selectedContentPanel == null) {
753 			return;
754 		}
755 		JNode node = selectedContentPanel.getNode();
756 		if (node.getParent() == null && treeRoot != null) {
757 			// node not register in tree
758 			node = treeRoot.searchNode(node);
759 			if (node == null) {
760 				LOG.error("Class not found in tree");
761 				return;
762 			}
763 		}
764 		TreeNode[] pathNodes = treeModel.getPathToRoot(node);
765 		if (pathNodes == null) {
766 			return;
767 		}
768 		TreePath path = new TreePath(pathNodes);
769 		tree.setSelectionPath(path);
770 		tree.makeVisible(path);
771 		tree.scrollPathToVisible(path);
772 		tree.requestFocus();
773 	}
