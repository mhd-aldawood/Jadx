632 		private String forwardJDWP() {
633 			int localPort = forwardTcpPort;
634 			String resultDesc = "";
635 			try {
636 				do {
637 					ForwardResult rst = device.forwardJDWP(localPort + "", pid);
638 					if (rst.state == 0) {
639 						forwardTcpPort = localPort;
640 						return "";
641 					}
642 					if (rst.state == 1) {
643 						if (rst.desc.contains("Only one usage of each socket address")) { // port is taken by other process
644 							if (localPort < 65536) {
645 								localPort++; // retry
646 								continue;
647 							}
648 						}
649 					}
650 					resultDesc = rst.desc;
651 					break;
652 				} while (true);
653 			} catch (Exception e) {
654 				LOG.error("JDWP forward error", e);
655 			}
656 			if (StringUtils.isEmpty(resultDesc)) {
657 				resultDesc = NLS.str("adb_dialog.forward_fail");
658 			}
659 			return resultDesc;
660 		}

663 		private void clearForward() {
664 			String jdwpPid = " jdwp:" + pid;
665 			String tcpPort = " tcp:" + forwardTcpPort;
666 			try {
667 				List<String> list = ADB.listForward(device.getDeviceInfo().adbHost,
668 						device.getDeviceInfo().adbPort);
669 				for (String s : list) {
670 					if (s.startsWith(device.getSerial()) && s.endsWith(jdwpPid) && !s.contains(tcpPort)) {
671 						String[] fields = s.split("\\s+");
672 						for (String field : fields) {
673 							if (field.startsWith("tcp:")) {
674 								try {
675 									device.removeForward(field.substring("tcp:".length()));
676 								} catch (Exception e) {
677 									LOG.error("JDWP remove forward error", e);
678 								}
679 							}
680 						}
681 					}
682 				}
683 			} catch (Exception e) {
684 				LOG.error("JDWP clear forward error", e);
685 			}
