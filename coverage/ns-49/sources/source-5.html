133 	private List<PkgNode> getPackageTree(List<String> names) {
134 		List<PkgNode> roots = new ArrayList<>();
135 		Set<String> nameSet = new HashSet<>();
136 		Map<String, List<PkgNode>> childMap = new HashMap<>();
137 		for (String name : names) {
138 			String parent = "";
139 			int last = 0;
140 			do {
141 				int pos = name.indexOf(".", last);
142 				if (pos == -1) {
143 					pos = name.length();
144 				}
145 				String fullName = name.substring(0, pos);
146 				if (!nameSet.contains(fullName)) {
147 					nameSet.add(fullName);
148 					PkgNode node = new PkgNode(fullName, name.substring(last, pos));
149 					if (!parent.isEmpty()) {
150 						childMap.computeIfAbsent(parent, k -> new ArrayList<>())
151 								.add(node);
152 					} else {
153 						roots.add(node);
154 					}
155 				}
156 				parent = fullName;
157 				last = pos + 1;
158 			} while (last < name.length());
159 		}
160 		addToParent(null, roots, childMap);
161 		return this.roots;
162 	}

164 	private PkgNode addToParent(PkgNode parent, List<PkgNode> roots, Map<String, List<PkgNode>> childMap) {
165 		for (PkgNode root : roots) {
166 			String tempFullName = root.getFullName();
167 			do {
168 				List<PkgNode> children = childMap.get(tempFullName);
169 				if (children != null) {
170 					if (children.size() == 1) {
171 						PkgNode next = children.get(0);
172 						next.name = root.name + "." + next.name;
173 						tempFullName = next.fullName;
174 						next.fullName = root.fullName;
175 						root = next;
176 						continue;
177 					} else {
178 						addToParent(root, children, childMap);
179 					}
180 				}
181 				if (parent == null) {
182 					this.roots.add(root);
183 				} else {
184 					parent.add(root);
185 				}

205 	private void walkTree(boolean findSelected, Consumer<PkgNode> consumer) {
206 		List<PkgNode> queue = new ArrayList<>(roots);
207 		for (int i = 0; i < queue.size(); i++) {
208 			PkgNode node = queue.get(i);
209 			if (findSelected && node.isSelected()) {
210 				consumer.accept(node);
211 			} else {
212 				if (!findSelected) {
213 					consumer.accept(node);
214 				}
215 				for (int j = 0; j < node.getChildCount(); j++) {
216 					queue.add((PkgNode) node.getChildAt(j));
