280 		searchEmitter = new SearchEventEmitter();
281 
282 		Flowable<String> textChanges = onTextFieldChanges(searchField);
283 		Flowable<String> searchEvents = Flowable.merge(textChanges, searchEmitter.getFlowable());
284 		searchDisposable = searchEvents
285 				.subscribeOn(Schedulers.single())
286 				.switchMap(text -> prepareSearch(text)
287 						.doOnError(e -> LOG.error("Error prepare search: {}", e.getMessage(), e))
288 						.subscribeOn(Schedulers.single())
289 						.toList()
290 						.toFlowable(), 1)
291 				.observeOn(SwingSchedulers.edt())

293 				.subscribe(this::processSearchResults);

296 	private Flowable<JNode> prepareSearch(String text) {
297 		if (text == null || options.isEmpty()) {
298 			return Flowable.empty();
299 		}
300 		// allow empty text for comments search
301 		if (text.isEmpty() && !options.contains(SearchOptions.COMMENT)) {
302 			return Flowable.empty();
303 		}
304 
305 		TextSearchIndex index = checkIndex();
306 		if (index == null) {
307 			return Flowable.empty();
308 		}
309 		LOG.debug("search event: {}", text);
310 		showSearchState();
311 		try {
312 			Flowable<JNode> result = index.buildSearch(text, options);
313 			if (searchField.getBackground() == SEARCHFIELD_ERROR_COLOR) {
314 				searchField.setBackground(searchFieldDefaultBgColor);
315 			}
316 			return result;
317 		} catch (SearchSettings.InvalidSearchTermException e) {
318 			searchField.setBackground(SEARCHFIELD_ERROR_COLOR);
319 			return Flowable.empty();
320 		}
321 	}

339 		return Flowable.<String>create(emitter -> {
340 			DocumentListener listener = new DocumentListener() {

348 					change();

359 			};

362 			emitter.setDisposable(new Disposable() {
363 				private boolean disposed = false;

373 					return disposed;
374 				}
375 			});
376 		}, BackpressureStrategy.LATEST)
377 				.debounce(300, TimeUnit.MILLISECONDS)
378 				.distinctUntilChanged();
